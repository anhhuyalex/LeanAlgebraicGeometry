% Introduction
% -----------------------------------------------------------------------

This chapter provides the mathematical roadmap for the formalisation of
Disegni's paper \emph{The universal $p$-adic Gross--Zagier formula}
(MR4493324, arXiv:2001.00045).  The principal goal is
\cref{thm:TheoremD}: the $p$-adic height of the universal Heegner class
over a Hida family equals the cyclotomic derivative of the $p$-adic
$L$-function.  The key specialisation to a fixed automorphic
representation is \cref{thm:TheoremB}.

\subsection{The $p$-adic Be\u\i linson--Bloch--Kato conjecture in analytic rank~$1$}

% SOURCE QUOTE PROOF: "We now show the existence of the filtration up  to further restricting the base. Fix a place $v\vert p$ of $F$, and let  $\det_{v}\colon G_{F_{v}}\to \OO(\X_{\G}^{\rm irr})^{\ts}$ be the character giving the action on $\det \cV_{\G,v}$. Let $\cV_{0,v}^{+}$ be the trivial  sheaf $\OO_{\X^{\rm irr}_{\G}}$ with $G_{F_{v}}$-action by the character \eqref{alpha gal}, $\cV_{0,v}:=\cV_{\G,v}$, $\cV_{0,v}^{-}:=(\cV_{0,v}^{+})^{-1}(\det_{v})$.  Finally, for $?=+, -. \emptyset$, let
$$\cW_{v}^{?}:= \Hom_{\OO_{\X'_{\G}}}(\cV_{0,v}^{+}, \cV_{0,v}^{?}).$$ Then for all $x\in \X_{\G}^{\rm cl}$, by Theorem \ref{galois for hilbert} we have exact sequences
\beq \lb{cW ex seq}
0\to \cW_{v,x}^{+}= \Q_{p}(x) \to \cW_{x}\to \cW_{v,x}^{-},
\eeq
which we  wish to extend to a neighbourhood of $\X_{\G}^{\cl}$." (read from references/MR4493324-padic-gross-zagier.tex)
\begin{proof}
  \uses{}
  \textit{Proof: see Disegni, §3.2 (Proposition 3.2.1).}
  The sheaf $\cV$ over $\cX$ is constructed in §3.2 from the completed homology module
  $\cM_{K^{p\prime}}$.  For each $w \mid p$, the ordinary filtration is extracted using
  Hida's $U_p$-operator: the universal ordinary character $\alpha^\circ_{|F_w^\times}
  \langle\,\cdot\,\rangle_{F_w}$ (arising from the eigenvalue of $U_p$ on the ordinary
  quotient) defines the line-bundle $\cV_w^+$ via its action on a rank-$1$ summand of
  $\cV_w$.  More precisely, one defines $\cW_v^? :=
  \mathrm{Hom}_{\scO_\cX}(\cV_{0,v}^+, \cV_{0,v}^?)$ and uses a weight argument at each
  classical point $x$ (where $H^0(F_v, \cW_{v,x}^-)= H^2(F_v, \cW_{v,x}^-)=0$) to
  propagate the splitting to an open neighbourhood of $\cX^{\mathrm{cl}}$.  At every
  $z \in \cX^{\mathrm{cl}}$, the specialisation $\cV_{w,z}^+$ recovers the ordinary
  sub-representation $V_w^+$ by the definition of ordinarity in §2.4.3.
\end{proof}


\begin{definition}[Bloch--Kato and \nek's Selmer groups]
  \label{def:SelmerGroup}
  \lean{MR4493324TheUniversalPAdicGrossZagierFormula.Basic.TODO.selmerGroup}
  \uses{def:GaloisRepresentation}
% SOURCE: [Disegni], §1.1.1 (read from references/MR4493324-padic-gross-zagier.tex)
% SOURCE QUOTE: "If $E$ is a number field and $V$ is a geometric $p$-adic  representation of its Galois group $G_{E}$, Bloch  and Kato \cite{bk} have proposed an analogue
% $$H^{1}_{f}(E,V)$$ of the Selmer group of $A$; it is an $L$-vector-subspace (where $L$ is the field of scalars for $V$) of the first Galois cohomology group of $V$, consisting of those classes satisfying certain local conditions. [...] \nek\ observed that a variation of Greenberg's definition works well in $p$-adic families, and developed this observation into the theory of Selmer complexes \cite{nek-selmer},  that provides the foundation for the  present work (\S~\ref{big sec: sel}). For nice $p$-adic families of $G_{E}$-representations, the theory allows to define groups
% $$\wtil{H}^{i}_{f}(E, V).$$
%  for all $i$."
  \textit{Source: Disegni, §1.1.1.}
  For a number field $E$ and a geometric $p$-adic $G_E$-representation
  $V$, the \emph{Bloch--Kato Selmer group} is $H^1_f(E,V)$, an
  $L$-subspace of $H^1(G_E, V)$ defined by local conditions.

  \nek's theory of Selmer complexes yields, for nice $p$-adic families
  of $G_E$-representations (in particular for the sheaf $\cV$ over
  $\cX$ with the local condition at $p$ coming from
  \cref{lem:OrdinaryReductionAtP}), the Selmer groups
  $\tH^i_f(E, V)$ for all $i \geq 0$.  When $V$ is ordinary and not
  exceptional, $\tH^1_f(E, V) = H^1_f(E, V)$.
\end{definition}

\begin{theorem}[Theorem A: $p$-adic Beilinson--Bloch--Kato in analytic rank $1$]
  \label{thm:TheoremA}
  \lean{MR4493324TheUniversalPAdicGrossZagierFormula.Basic.TODO.theoremA}
  \uses{def:SelmerGroup, def:HeightPairing, def:PAdicLFunction,
        def:OrdinaryAutomorphicRep, thm:TheoremB}
% SOURCE: [Disegni], §1.1, Theorem A (BK thm) (read from references/MR4493324-padic-gross-zagier.tex)
% SOURCE QUOTE: "\begin{theoA}\label{BK thm} Let $\pi_{0}$ be a Hilbert modular form over $L$ of weight $\uw$, and let $\chi$ be a Hecke character of  $E$ over $L$ of weight $\ul$.

% \begin{enumerate}

% \item \lb{2fou}

%   \begin{enumerate}

% \item
% let $R_{T} \in \OO_{L}\hat{\ot}_{\Z_{p}}\Gamma_{F} $ be the regulator of the height pairing \eqref{ht V}  on $\wtil{H}^{1}_{f}(E, T)\ts\wtil{H}^{1}_{f}(E, T^{*}(1))$. Then
% $$ \calL_{p}'(V_{(\pi, \chi)}, 0) \succeq_{\Z_{p}} R_{T}\cdot | \wtil{H}^{2}_{f}(E, T)_{\rm tors}| $$
% in $L\hat{\ot}\Gamma_{F} $.
% \end{enumerate}
% \end{enumerate}
% \end{theoA}"
  \textit{Source: Disegni, Theorem~A (§1.1).}
  Let $\pi_0$ be a Hilbert modular form over $L$ of weight
  $\underline{w}$ and let $\chi$ be a Hecke character of $E$ over $L$ of
  weight $\underline{l}$.  Let $V := V_{\pi_0|G_E} \otimes V_\chi$.
  Suppose that:
  \begin{itemize}
    \item[(wt)] $|l_\sigma| < w_\sigma$ for all $\sigma \colon F
          \hookrightarrow L$;
    \item[(sd)] $\omega_{\pi_0}\omega_\chi = 1$ (which implies
          $w + l = 0$);
    \item[$(\varepsilon)$] $\varepsilon(V) = -1$;
    \item[(ord)] $\pi_0$ is ordinary;
    \item[(n-exc)] $V$ is \emph{not exceptional}: for no place
          $w \mid v \mid p$ of $E$ is
          $V_w^- := V^-_{\pi_0,v|G_{E_w}} \otimes \chi_w$ the trivial
          representation.
  \end{itemize}
  Then:
  \begin{enumerate}
    \item We have
          \[
            \cL_p'(V_{(\pi,\chi)}, 0) \neq 0
            \;\Longrightarrow\;
            \dim_L \tH^1_f(E, V) \geq 1,
          \]
          and one can exhibit an explicit nonzero element of
          $\tH^1_f(E, V) = H^1_f(E, V)$ whose $p$-adic height is also
          non-zero.
    \item Let $T \subset V$ be a stable lattice.  If
          $\cL_p'(V_{(\pi,\chi)}, 0) \neq 0$ and moreover the conditions
          of Fouquet~\cite[Theorem~B(i)]{fouquet} are satisfied, then:
          \begin{enumerate}
            \item $\dim_L \tH^1_f(E, V) = 1$;
            \item letting $R_T \in \scO_L\,\hat\otimes_{\bZ_p}\,\GF$ be
                  the regulator of the height pairing on
                  $\tH^1_f(E, T) \otimes \tH^1_f(E, T^*(1))$, one has
                  \[
                    \cL_p'(V_{(\pi,\chi)}, 0)
                    \succeq_{\bZ_p}
                    R_T \cdot |\tH^2_f(E, T)_{\mathrm{tors}}|
                  \]
                  in $L\,\hat\otimes\,\GF$, where $m_1 \succeq_A m_2$
                  means $m_1 \subseteq m_2$ as $A$-submodules.
          \end{enumerate}
  \end{enumerate}
\end{theorem}

% SOURCE QUOTE PROOF: "Part 1    will be an immediate consequence of Theorem \ref{GZ thm},  the Jacquet--Langlands correspondence, and the observation following \eqref{eps v} below. [...] Part 2 follows from invoking the results of Fouquet in \cite{fouquet}, that generalise the bounds on Tate--Shafarevich groups of elliptic curves obtained by Kolyvagin using the methods of Euler systems." (read from references/MR4493324-padic-gross-zagier.tex)
\begin{proof}
  \uses{thm:TheoremB}
  \textit{Proof: see Disegni, §1.1 (discussion following Theorem~A).}
  Part~1 is an immediate consequence of \cref{thm:TheoremB}, the
  Jacquet--Langlands correspondence, and the local epsilon-factor
  observation: when $\varepsilon(V) = -1$ and $\Pi$ is ordinary,
  locally distinguished and non-exceptional, the height of the Heegner
  class $P_\Pi$ equals $e_{p\infty}(V)^{-1}\,\cL_p'(V,0)\,Q(\cdot)$,
  which is non-zero exactly when $\cL_p'(V,0) \neq 0$, exhibiting an
  explicit class in $\tH^1_f(E,V) = H^1_f(E,V)$ of non-zero height,
  whence $\dim_L \tH^1_f(E,V) \geq 1$.  Part~2 follows by invoking
  Fouquet's Euler-system bounds \cite{fouquet}, which generalise
  Kolyvagin's bounds on Tate--Shafarevich groups of elliptic curves and
  pin $\dim_L \tH^1_f(E,V) = 1$ together with the stated divisibility.
\end{proof}

% SOURCE QUOTE: "Let $\B$ be a quaternion algebra over $F_{\A}$ (where $\A$ denotes the ad\`eles of $\Q$) with ramification set $\Sigma\sqcup \{v\vert \infty\} $ satisfying $|\Sigma| \equiv [F:\Q] -1\pmod{2}$.   Let $\bB$ be a quaternion algebra over $F_{\bA}$ with ramification
  set $\Sg \sqcup \{v \mid \infty\}$ satisfying
  $|\Sg| \equiv [F:\bQ] - 1 \pmod{2}$.  Then

  $E_{\bA} \hookrightarrow \bB$, of which we fix one, yielding
  $\mathrm{e} \colon H \hookrightarrow G$.

\begin{definition}[Locally distinguished]
  \label{def:LocallyDistinguished}
  \lean{MR4493324TheUniversalPAdicGrossZagierFormula.Basic.TODO.locallyDistinguished}
  \uses{def:AutomorphicRepresentation, def:GaloisRepresentation}
% SOURCE: [Disegni], §1.3.4 (read from references/MR4493324-padic-gross-zagier.tex)
% SOURCE QUOTE: " \begin{defi}\lb{loc dist}
%  We say that $\Pi$ is \emph{locally distinguished by $\H'$}, or simply \emph{locally distinguished}, if it satisfies conditions $(\eps_{v}) $ for all $v$.
%  \end{defi}"
  \textit{Source: Disegni, §1.3.4, Definition (loc dist).}
  For each place $v$ of $F$, define the local epsilon factor
  \[
    \varepsilon_v^G(V) := \varepsilon(V_v)\,\chi_v(-1)\,\eta_v(-1)\,

  for all places $v$.
\end{definition}

\begin{definition}[Global toric period pairing $Q$]

  normalised by the local $L$-factor $\mathscr{L}(V_{(\pi,\chi),v},0)$
  and the Haar measure $dt_v$.  Choosing measures with
  $\mathrm{vol}(H'(\bQ)\backslash H'(\bA), dt) = 1$, the \emph{global
  toric period pairing} is
  \[
    Q\!\left(\frac{f_1 \otimes f_2}{f_3 \otimes f_4}\right)
    := \prod_v Q_{v,dt_v}\!\left(\frac{f_{1,v} \otimes f_{2,v}}
    {f_{3,v} \otimes f_{4,v}}\right)
  \]
  for $f_1 \in \Pi_{H'}$, $f_2 \in \Pi^\vee_{H'}$, $f_3 \in \Pi$,
  $f_4 \in \Pi^\vee$ with $\prod_v (f_{3,v}, f_{4,v})_v \neq 0$.
\end{definition}

  $G(R) = (B \otimes R)^\times$, and fix an embedding
  $\mathrm{e} \colon E \hookrightarrow B$; here all associated Shimura
  varieties are $0$-dimensional.  The normalised fundamental class of
  $Y$ yields an $H'(\bA)$-invariant functional $P_\Pi \colon \Pi \to L$
  and, over a locally distinguished Hida family $\cX$ for
  $(G \times H)'$ (isomorphic via Jacquet--Langlands to a family $\cX_0$
  for $(G_0 \times H)'$), an $H'(\bA^{p\infty})$-invariant
  $\scO_\cX$-linear functional
  $\cP \colon \Pi^{K^p,\ord}_{H'_\Sg} \to \scO_\cX$ interpolating the
  $P^\ord_{\Pi_z}$ at all $z \in \cX^{\mathrm{cl}}$ satisfying~(wt).

  Abbreviate $\Pi^{(\iota)} := \Pi^{K^{p\prime},\ord,(\iota)}_{H'_\Sg}$,
  $\scO := \scO_\cX$, $\cK := \cK_\cX$.  There exists an open subset
  $\cX' \subset \cX$ containing $\cX^{\mathrm{cl}}$ such that
  \[
    \frac{\cP(f_1) \cdot \cP^\iota(f_2)}{((f_3, f_4))}
    =
    \cL_p(\cV^\sharp)_{|\cX}
    \cdot
    \cQ\!\left(\frac{f_1 \otimes f_2}{f_3 \otimes f_4}\right),
  \]
  an equality of $\cK$-valued $\scO$-linear functionals on
  $(\Pi \otimes_\scO \Pi^\iota) \otimes_{\scO^\times}
  (\Pi \otimes_\scO \Pi^\iota)^{\times,-1}$.
\end{theorem}
