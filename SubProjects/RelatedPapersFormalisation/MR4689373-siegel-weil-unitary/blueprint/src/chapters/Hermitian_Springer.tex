\chapter{Hermitian Springer Theory}
\label{chap:hermitian-springer}

This chapter formalizes Phase~P3a of the project: the Hermitian analogue of Laumon's
Springer sheaf on \(\mathrm{Coh}_d(X)\). The objects here are the moduli stack
\(\Herm_{2d}(X'/X)\) of Hermitian torsion sheaves, its Grothendieck--Springer resolution
\(\widetilde{\Herm}_{2d}(X'/X)\), and the resulting Hermitian Springer sheaf
\(\mathrm{Spr}^{\mathrm{Herm}}_{2d}\) with its canonical \(W_d\)-action and
decomposition into isotypical summands. We follow \S\S3--4 of Feng--Yun--Zhang
[MR4689373].

\section{Springer theory for Hermitian torsion sheaves}

\subsection{Local geometry of $\Herm_{d}$}

\begin{definition}[Moduli stack of Hermitian torsion sheaves]
  \label{def:herm-torsion-stack}
  \lean{HSW.HermTorsionStack}
  \uses{def:etale-cover, def:serre-dual-vb, def:base-curve}
  % SOURCE: FYZ-HigherSiegelWeil.tex, §3.1
  % SOURCE QUOTE: "\(\Herm_{2d}(X'/X)\) parameterizes pairs \((\cQ,\iota)\) where
  %   \(\cQ\) is a torsion sheaf on \(X'\) of length \(2d\) and
  %   \(\iota:\cQ\xrightarrow{\sim}\sigma^*\cQ^\vee\) is a Hermitian structure"
  \textit{Source: Feng--Yun--Zhang [MR4689373], \S3.1.}
  For \(d \geq 0\), the moduli stack \(\Herm_{2d}(X'/X)\) parameterizes pairs
  \((\cQ, \iota)\) where:
  \begin{enumerate}
    \item \(\cQ\) is a coherent torsion sheaf on \(X'\) of \(\mathcal{O}_{X'}\)-length
      \(2d\) (i.e., a rank-0 coherent sheaf supported at finitely many closed points,
      with total length \(2d\));
    \item \(\iota : \cQ \xrightarrow{\;\sim\;} \sigma^*\cQ^\vee\) is a
      \emph{Hermitian structure}, where \(\cQ^\vee = \underline{\mathrm{Ext}}^1(\cQ,
      \om_{X'})\) is the Serre dual of \(\cQ\) (Definition~\ref{def:serre-dual-vb})
      and \(\sigma\) is the involution of \(X'/X\) (Definition~\ref{def:etale-cover}).
  \end{enumerate}
  The stack \(\Herm_{2d}(X'/X)\) is an algebraic stack locally of finite type over
  \(\mathbf{F}_q\), stratified by the isomorphism type of \(\cQ\) as a
  \(\sigma\)-equivariant torsion sheaf.
\end{definition}

\subsection{The Hermitian Springer sheaf}

\begin{definition}[Hermitian Springer resolution]
  \label{def:herm-springer-resolution}
  \lean{HSW.HermSpringerResolution}
  \uses{def:herm-torsion-stack, def:etale-cover}
  % SOURCE: FYZ-HigherSiegelWeil.tex, §ss:Herm Spr (The Hermitian Springer sheaf)
  % SOURCE QUOTE: "$\wt\Herm_{d}(X'/X)$ ... classifying $(\cQ,h)\in \Herm_{d}(X'/X)$ together with a full flag
  %   $0\subset \cQ_{1}\subset \cdots \subset \cQ_{i}\subset \cdots \subset \cQ_{d-1}=\cQ_{1}^{\bot}\subset \cQ_{d}=\cQ$,
  %   where $\cQ_{i}$ has length $i$ and $\cQ_{d-i}=\cQ_{i}^{\bot}$"
  \textit{Source: Feng--Yun--Zhang [MR4689373], \S``The Hermitian Springer sheaf''.}
  The \emph{Hermitian Springer resolution} is the algebraic stack
  \(\widetilde{\Herm}_{2d}(X'/X)\) parameterizing \emph{full} flags of sub-sheaves
  \[
    0 = \cQ_0 \subset \cQ_1 \subset \cdots \subset \cQ_i \subset \cdots
      \subset \cQ_{2d} = \cQ
  \]
  in which each step has length \(1\), i.e.\ \(\cQ_i\) has length \(i\) for
  \(i = 0, 1, \ldots, 2d\), together with the Hermitian structure \(\iota\) on
  \(\cQ = \cQ_{2d}\), subject to the isotropy constraint
  \(\cQ_{2d-i} = \cQ_i^{\bot}\) for all \(i\) (orthogonal complement under the
  Hermitian pairing). The forgetful map
  \[
    \tilde\pi : \widetilde{\Herm}_{2d}(X'/X) \to \Herm_{2d}(X'/X),
    \quad (\cQ_\bullet) \mapsto \cQ_{2d},
  \]
  is a proper representable morphism of algebraic stacks.
\end{definition}

\begin{definition}[Hermitian Springer sheaf]
  \label{def:springer-sheaf-herm}
  \lean{HSW.HermSpringerSheaf}
  \uses{def:herm-springer-resolution, def:herm-torsion-stack}
  % SOURCE: FYZ-HigherSiegelWeil.tex, §3.4–4.1
  % SOURCE QUOTE: "the perverse sheaf \(\mathrm{Spr}^\Herm_{2d}:=\tilde\pi_*\Ql\)
  %   ... carries a canonical action of the group
  %   \(W_d=(\BZ/2\BZ)^d\rtimes S_d\)"
  \textit{Source: Feng--Yun--Zhang [MR4689373], \S3.4--4.1.}
  The \emph{Hermitian Springer sheaf} is the direct image perverse sheaf
  \[
    \mathrm{Spr}^{\mathrm{Herm}}_{2d} := \tilde\pi_*\, \Ql
    \;\in\; \mathrm{Perv}(\Herm_{2d}(X'/X),\, \Ql),
  \]
  where \(\Ql = \overline{\mathbf{Q}}_\ell\) is the \(\ell\)-adic coefficient field.
  By the Decomposition Theorem (Beilinson--Bernstein--Deligne--Gabber), since
  \(\tilde\pi\) is proper, \(\tilde\pi_*\Ql\) is semisimple.
  The Weyl group \(W_d = (\mathbf{Z}/2\mathbf{Z})^d \rtimes S_d\) acts on
  \(\mathrm{Spr}^{\mathrm{Herm}}_{2d}\) by Hecke correspondences:
  \begin{itemize}
    \item the \(S_d\)-action permutes the steps of the flag \((\cQ_\bullet)\);
    \item the \((\mathbf{Z}/2\mathbf{Z})^d\)-action applies the Hermitian involution
      \(\iota\) on individual graded pieces \(\cQ_i/\cQ_{i-1}\).
  \end{itemize}
  This is the Hermitian analogue of Laumon's Springer sheaf on \(\mathrm{Coh}_d(X)\).
\end{definition}

\subsection{The action of $W_{d}$}

\begin{theorem}[Decomposition of the Hermitian Springer sheaf]
  \label{thm:springer-decomp}
  \lean{HSW.springerDecomp}
  \uses{def:springer-sheaf-herm, def:herm-torsion-stack}
  % SOURCE: FYZ-HigherSiegelWeil.tex, §4.2–4.3
  % SOURCE QUOTE: "we ... decompose \(\mathrm{Spr}^\Herm_{2d}\) into isotypical
  %   summands \(\bigoplus_\chi V_\chi\otimes\mathrm{IC}_\chi\)"
  \textit{Source: Feng--Yun--Zhang [MR4689373], \S4.2--4.3.}
  There is a canonical \(W_d\)-equivariant decomposition
  \[
    \mathrm{Spr}^{\mathrm{Herm}}_{2d}
    \;\cong\;
    \bigoplus_{\chi \in \mathrm{Irr}(W_d)}
    V_\chi \otimes_{\Ql} \mathrm{IC}_\chi
  \]
  in \(\mathrm{Perv}(\Herm_{2d}(X'/X), \Ql)\), where:
  \begin{itemize}
    \item \(\mathrm{Irr}(W_d)\) is the set of irreducible \(\Ql\)-representations of
      \(W_d\), parameterized by bipartitions \((\lambda, \mu)\) with
      \(|\lambda| + |\mu| = d\);
    \item \(V_\chi\) is the underlying \(\Ql\)-vector space of the irreducible
      representation \(\chi\) of \(W_d\);
    \item \(\mathrm{IC}_\chi\) is the intersection cohomology complex on
      \(\Herm_{2d}(X'/X)\) associated to the \(\chi\)-isotypical local system on
      the open stratum.
  \end{itemize}
  The multiplicity of \(\mathrm{IC}_\chi\) in \(\mathrm{Spr}^{\mathrm{Herm}}_{2d}\)
  is \(\dim_{\Ql} V_\chi\).
\end{theorem}

\begin{proof}
  \uses{def:springer-sheaf-herm, def:herm-torsion-stack}
  By the Decomposition Theorem (BBD), since
  \(\tilde\pi : \widetilde{\Herm}_{2d} \to \Herm_{2d}\) is proper,
  \(\tilde\pi_*\Ql\) is a semisimple object in the derived category. Its
  perverse cohomology sheaves are semisimple perverse sheaves; since \(\tilde\pi\)
  is of relative dimension \(0\) generically, the main constituent is
  \(\mathrm{Spr}^{\mathrm{Herm}}_{2d} = \tilde\pi_*\Ql\) itself.
  The \(W_d\)-equivariance of \(\mathrm{Spr}^{\mathrm{Herm}}_{2d}\) (from
  Definition~\ref{def:springer-sheaf-herm}) determines the decomposition into
  isotypical summands: each summand \(\mathrm{IC}_\chi\) is the IC sheaf of the
  \(\chi\)-isotypical local system on the open stratum of \(\Herm_{2d}\) (the stratum
  of maximally split Hermitian torsion sheaves), identified using the combinatorial
  structure of the \(W_d\)-action.
\end{proof}
