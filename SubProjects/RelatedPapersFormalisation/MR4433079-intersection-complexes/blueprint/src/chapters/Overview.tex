\chapter{Overview}
\label{chap:overview}

% archon:covers MR4433079IntersectionComplexesAndUnramifiedLFactors.lean MR4433079IntersectionComplexesAndUnramifiedLFactors/Basic.lean

This chapter establishes the complete mathematical roadmap for formalizing
Sakellaridis--Wang, \textit{Intersection complexes and unramified $L$-factors}
(arXiv:2009.03943, MR4433079).
The overall goal is to prove two main theorems
(\cref{thm:IC-affine-closure} and \cref{thm:IC-general}):
for an affine spherical \( G \)-variety \( X \) satisfying a ``type-T'' dual group condition,
the IC function on the arc space \( \Lplus X \) is expressed as a ratio of local
unramified \( L \)-values.  The dependency graph is rooted at these two theorems,
supported by a Kashiwara crystal (\cref{thm:crystal}), a semi-smallness theorem
(\cref{thm:semismall}), dimension estimates on central fibers
(\cref{thm:dimension-estimates}), and a nearby-cycles commutativity result
(\cref{thm:nearby-asymptotics}).

\begin{definition}[Reductive group and Borel data]
  \label{def:reductive-group-data}
  \lean{MR4433079IntersectionComplexesAndUnramifiedLFactors.ReductiveGroupData}
  \uses{}
% SOURCE: [Sakellaridis--Wang 2020], Notation Table §1.5 and §1.1
%         (read from references/MR4433079-intersection-complexes.tex)
% SOURCE QUOTE:
%   ``$G$ & a connected reductive group over $k$.
%   $T$ & the (abstract) Cartan of $G$, i.e., the reductive quotient of any Borel
%   subgroup. We sometimes fix a splitting $T\hookrightarrow B\hookrightarrow G$ of
%   the abstract Cartan into a Borel subgroup.
%   $W$ & the (abstract) Weyl group of $G$.
%   $\check \Lambda_G$ ($\Lambda_G$) & the coweight (resp.~weight) lattice of $T$. The index
%   $G$ will often be omitted.
%   $\check\Lambda^+_G$ ($\Lambda_G^+,\, \ch\Lambda^-_G$) & The monoid of dominant coweights
%   (resp., dominant weights, antidominant coweights).
%   $\ch\Lambda^\pos_G$ ($\Lambda^\pos_G \subset \Lambda_G$) & The monoid generated by the
%   non-negative \emph{integral} span of the positive coroots (resp.~roots) in $\ch\Lambda_G$.
%   $\check\Delta_G$ ($\Delta_G$) & the set of simple coroots (resp.~roots) of $G$.
%   $\ch\lambda \ge \ch\mu$ & For $\ch\lambda,\ch\mu \in \ch\Lambda_G$, this means that
%   $\ch\lambda - \ch\mu \in \ch\Lambda^\pos_G$.''
  \textit{Source: Sakellaridis--Wang (2020), Notation Table §1.5.}

  Let \( k \) be an algebraically closed field and \( G \) a connected reductive algebraic
  group over \( k \), equipped with a fixed splitting
  \( T \hookrightarrow B \hookrightarrow G \)
  of the abstract Cartan \( T = B/N \) into a Borel subgroup \( B \) with unipotent radical
  \( N \).
  Record the following data:
  \begin{itemize}
    \item The cocharacter lattice \( \checkLambda_G \) and character lattice
          \( \Lambda_G \) of \( T \).
    \item The dominant coweight monoid \( \checkLambda_G^+ \), the antidominant coweight
          monoid \( \checkLambda_G^- \), and the positive coroot monoid
          \( \checkLambda_G^{\mathrm{pos}} \) (the non-negative integral span of positive
          coroots).
    \item The set of simple coroots \( \check\Delta_G \) and simple roots \( \Delta_G \),
          and the Weyl group \( W \).
    \item The partial order: for \( \checklambda, \check\mu \in \checkLambda_G \),
          \( \checklambda \ge \check\mu \) iff
          \( \checklambda - \check\mu \in \checkLambda_G^{\mathrm{pos}} \).
    \item The Langlands dual group \( \checkG \) (the connected reductive group whose
          coweights are the weights of \( G \)).
  \end{itemize}
\end{definition}

\begin{definition}[Type-T (dual group) condition]
  \label{def:type-T-condition}
  \lean{MR4433079IntersectionComplexesAndUnramifiedLFactors.TypeTCondition}
  \uses{def:spherical-variety, def:reductive-group-data}
% SOURCE: [Sakellaridis--Wang 2020], §1.1, equation (equation-typeT)
%         (read from references/MR4433079-intersection-complexes.tex)
% SOURCE QUOTE:
%   ``$B$ acts freely on $X^\circ$, and for every simple root $\alpha$, if $P_\alpha$ is the
%   parabolic generated by $B$ and the root space of $-\alpha$, the stabilizer of a point in
%   the open $P_\alpha$-orbit $X^\circ P_\alpha$ is a torus (necessarily one-dimensional).''
  \textit{Source: Sakellaridis--Wang (2020), §1.1, equation~(1.1) (type-T condition).}

  The \emph{type-T condition} on an affine spherical \( G \)-variety \( X \) with open
  \( B \)-orbit \( X^\circ \) is:
  \begin{enumerate}
    \item \( B \) acts freely on \( X^\circ \); and
    \item for every simple root \( \alpha \), if \( P_\alpha \) is the minimal parabolic
          generated by \( B \) and the root space \( \mathfrak{g}_{-\alpha} \), the
          stabilizer in \( P_\alpha \) of a point in the open \( P_\alpha \)-orbit
          \( X^\circ P_\alpha \) is a (necessarily one-dimensional) torus.
  \end{enumerate}
  This condition is equivalent to saying that the Gaitsgory--Nadler--Sakellaridis--Venkatesh
  dual group \( \check{G}_X \) equals the Langlands dual group \( \checkG \) of \( G \).
  In particular, it implies \( T_X = T \), so we may canonically identify \( X \GIT N \) as
  an affine embedding of \( T \).
\end{definition}

\begin{definition}[Arc space]
  \label{def:arc-space}
  \lean{MR4433079IntersectionComplexesAndUnramifiedLFactors.ArcSpace}
  \uses{def:spherical-variety}
% SOURCE: [Sakellaridis--Wang 2020], §1.1, opening paragraph
%         (read from references/MR4433079-intersection-complexes.tex)
% SOURCE QUOTE:
%   ``The formal arc space $\msf L^+X$ is the infinite-dimensional scheme that represents
%   the functor $R\mapsto X(R\tbrac t)$, and it is singular, if $X$ is. However, its
%   singularities at ``generic'' $\mathbb F$-points (namely, arcs $\Spec \mathbb F\tbrac t
%   \to X$ which generically lie in the smooth locus $X^{\text{sm}}$) are of finite type,
%   according to the theorem of Grinberg--Kazhdan and Drinfeld, and this allows one to
%   define an \emph{IC function} \cite{BNS}, that is, the function that should correspond
%   under Frobenius trace to the ``intersection complex'' of $\msf L^+X$.''
  \textit{Source: Sakellaridis--Wang (2020), §1.1.}

  The \emph{formal arc space} \( \Lplus X \) is the ind-scheme representing the functor
  \( R \mapsto X(R\llbracket t \rrbracket) \).
  It is generally infinite-dimensional and singular.
  However, the Grinberg--Kazhdan--Drinfeld theorem asserts that the singularities at
  \emph{generic} points (arcs whose generic fiber lies in the smooth locus \( X^{\mathrm{sm}} \))
  are of finite type, enabling the definition of an intersection complex on formal neighborhoods.
  The Lean formalization of this object requires pro-scheme or ind-scheme infrastructure.
\end{definition}

\begin{definition}[IC function]
  \label{def:IC-function}
  \lean{MR4433079IntersectionComplexesAndUnramifiedLFactors.ICFunction}
  \uses{def:arc-space, def:spherical-variety, def:reductive-group-data}
% SOURCE: [Sakellaridis--Wang 2020], §1.1, second paragraph
%         (read from references/MR4433079-intersection-complexes.tex)
% SOURCE QUOTE:
%   ``This is a function $\Phi_0$ on $X(\mathfrak o)\cap X^{\text{sm}}(F)$ (where
%   $\mathfrak o=\mathbb F\tbrac t$ and $F = \mathbb F\lbrac t$), and it was conjectured in
%   \cite{SaRS} (before the rigorous definition of this function was available) that it is
%   related to special values of $L$-functions.''
%
%   ``We would normalize such a sheaf to be constant in degree zero over the arc space of the
%   smooth locus $\msf L^+X^{\text{sm}}$, and we would normalize the intersection complex of a
%   substratum $S\subset \msf L^+X$ of codimension $d$ to be $\ol\bbQ_\ell(\frac{d}{2})[d]$ on
%   its smooth locus, where $\ol\bbQ_\ell(\frac{1}{2})$ denotes a chosen square root of the
%   cyclotomic Tate twist.''
  \textit{Source: Sakellaridis--Wang (2020), §1.1.}

  Let \( k = \overline{\mathbb{F}} \) be the algebraic closure of a finite field,
  \( \mathfrak{o} = k\llbracket t \rrbracket \), \( F = k(\!(t)\!) \).
  The \emph{IC function} \( \Phi_0 \) is a \( \overline{\mathbb{Q}}_\ell \)-valued function
  on \( X(\mathfrak{o}) \cap X^{\mathrm{sm}}(F) \), defined as the Frobenius trace of the
  intersection complex of (finite-dimensional formal models of) \( \Lplus X \).
  The normalization convention is:
  \begin{itemize}
    \item On the smooth locus \( \Lplus X^{\mathrm{sm}} \), the IC sheaf is
          \( \overline{\mathbb{Q}}_\ell \) in degree zero.
    \item On a smooth stratum of codimension \( d \) in \( \Lplus X \), the IC sheaf is
          \( \overline{\mathbb{Q}}_\ell(\tfrac{d}{2})[d] \), where
          \( \overline{\mathbb{Q}}_\ell(\tfrac{1}{2}) \) is a chosen square root of the
          cyclotomic Tate twist.
  \end{itemize}
  The concrete computation of \( \Phi_0 \) proceeds via finite-dimensional Zastava models
  (\cref{def:zastava-model,def:global-model}).
\end{definition}

  There is a symplectic representation \( \rho_X \) of the \( L \)-group,
  \( \rho_X \colon {}^{L}G = \checkG \rtimes \langle \Fr \rangle \to \mathrm{GL}(V_X) \),
  and a \( {}^{L}T \)-stable polarization \( V_X = V_X^+ \oplus V_X^- \), such that the
  multiset \( \mfB^+ \) of \( \checkT \)-weights of \( V_X^+ \) belongs to \( \mfc_X \),
  and the pushforward of the IC function satisfies
  \begin{equation}
    \label{equation-pushforward-IC}
    (\eta\delta)^{1/2} \cdot \pi_! \Phi_0
      = \Bigl(\mathrm{tr}_{\checkT}\!\bigl(\Fr,\, \Sym^\bullet(\check{\mathfrak{n}}(1))\bigr)
        \Bigr)^{-1}
        \cdot \mathrm{tr}_{\checkT}\!\bigl(\Fr,\, \Sym^\bullet(V_X^+)\bigr).
  \end{equation}
  Here, for a representation \( V \) of \( {}^{L}T = \checkT \rtimes \langle \Fr \rangle \),
  the expression \( \mathrm{tr}_{\checkT}(\Fr, V) \) denotes the function on
  \( \checkLambda^{\Fr} \) whose value on \( \checklambda \) is the trace of geometric
  Frobenius \( \Fr \) on the \( (\checkT, \checklambda) \)-eigenspace of \( V \).

  \( \mathbb{F} \), the permutation action of \( \Fr \) is trivial and
  \eqref{equation-pushforward-IC} reads
  \begin{equation}
    \label{equation-pushforward-IC-expl}
    (\eta\delta)^{1/2}\, \pi_! \Phi_0
      = \frac{\prod_{\check\alpha \in \check\Phi^+} (1 - q^{-1} e^{\check\alpha})}
             {\prod_{\checklambda \in \mfB^+} (1 - q^{-1/2} e^{\checklambda})},
  \end{equation}
  where \( \mfB^+ \) is the multiset of weights and \( e^{\checklambda} \) denotes the
  characteristic function of the \( T(\mfo) \)-orbit of \( t^{\checklambda} \).
\end{conjecture}

\begin{definition}[Global model]
  \label{def:global-model}
  \lean{MR4433079IntersectionComplexesAndUnramifiedLFactors.GlobalModel}

% SOURCE QUOTE:

%

  \textit{Source: Sakellaridis--Wang (2020), §1.3.}

  Fix a smooth projective curve \( C \) over \( k \).
  For an algebraic stack \( \mathscr{X} \) and open substack \( \mathscr{X}^\circ \),
  \( \Maps_{\mathrm{gen}}(C, \mathscr{X} \supset \mathscr{X}^\circ) \) denotes the
  prestack of maps \( C \times S \to \mathscr{X} \) (for test schemes \( S \)) that
  generically land in \( \mathscr{X}^\circ \).

  The \emph{global model} for the arc space of \( X \) is the Artin stack
  \[
    \sM_X = \Maps_{\mathrm{gen}}(C,\; X/G \supset X^\bullet/G).
  \]
  It lives over \( \Bun_G \), the moduli stack of \( G \)-bundles on \( C \).
  The restriction of a \( k \)-point of \( \sM_X \) to the formal neighborhood of
  a point \( v \in |C| \) gives an element of
  \( (X^\bullet(F_v) \cap X(\mathfrak{o}_v))/G(\mathfrak{o}_v) \).
\end{definition}

\begin{definition}[Central fibers]
  \label{def:central-fibers}
  \lean{MR4433079IntersectionComplexesAndUnramifiedLFactors.CentralFibers}

% SOURCE: [Sakellaridis--Wang 2020], §1.3, §1.4, and §5
%         (read from references/MR4433079-intersection-complexes.tex)
% SOURCE QUOTE:
%   ``Hence for now let us focus on the case $X=\overline{X^\bullet}^\aff$, assuming that
%   $\mathfrak c_X$ is free. In that case, the maximal possible dimension for the central
%   fiber $\msf Y^{\check\lambda}$ will be called the \emph{critical dimension}, and is equal
%   to $\frac{1}{2} (\len(\check\lambda)-1),$ where $\len(\check\lambda) := \sum_i m_i$, for
%   $\check\lambda\in \mathfrak c_X$ written uniquely as $\sum_i m_i \ch\nu_i$ in terms of
%   our basis of $\mathfrak c_X$.''
%
%   ``There is another way to understand the central fibers $\msf Y^{\check\lambda}$: as
%   subsets of the affine Grassmannian of $G$. Let us fix the point $v \in C$ that we take
%   central fibers with respect to. Then a $k$-point of $\msf Y^{\ch\lambda}$ is a map
%   $C \to X/B$ such that $C\sm v$ is sent to $X^\circ/B=\pt$.''
  \textit{Source: Sakellaridis--Wang (2020), §1.3, §1.4, and §5.}

  Fix a point \( v \in |C| \).  For \( \checklambda \in \mfc_X \), the
  \emph{central fiber} \( \msf{Y}^{\checklambda} \) is the fiber of
  \( \pi \colon \sY_X \to \sA \) over the ``diagonal'' stratum corresponding to the
  \( \mfc_X \)-valued divisor \( \checklambda \cdot [v] \) concentrated at \( v \).
  Concretely, \( \msf{Y}^{\checklambda}(k) \) consists of maps
  \( C \to X/B \) sending \( C \setminus \{v\} \) to \( X^\circ/B \).

  Via the identification with semi-infinite orbits in the affine Grassmannian
  (\cref{def:MV-cycles}):
  \[
    \msf{Y}^{\checklambda}_{\mathrm{red}}
      = \bigl(\msf{S}^{\checklambda} \times_{\Lplus X/\Lplus G}
              \Lplus X/\Lplus G \bigr)_{\mathrm{red}}.
  \]
  When \( X = X^{\mathrm{can}} \) and \( \mfc_X \cong \mathbb{N}^r \) is free, the
  \emph{critical dimension} of \( \msf{Y}^{\checklambda} \) is
  \[
    \tfrac{1}{2}\bigl(\len(\checklambda) - 1\bigr),
    \quad \text{where}\quad
    \len(\checklambda) := \sum_i m_i
    \quad \text{for } \checklambda = \sum_i m_i \checknu_i.
  \]
  The set \( \mfB_{X^\bullet}^+ \) of irreducible components of central fibers of critical
  dimension is the key geometric input for the crystal structure
  (\cref{thm:crystal}).
\end{definition}

\begin{definition}[Semi-infinite orbits and MV cycles]
  \label{def:MV-cycles}
  \lean{MR4433079IntersectionComplexesAndUnramifiedLFactors.MVCycles}
  \uses{def:reductive-group-data, def:arc-space}
% SOURCE: [Sakellaridis--Wang 2020], §1.4
%         (read from references/MR4433079-intersection-complexes.tex)
% SOURCE QUOTE:
%   ``Semi-infinite orbits have an important meaning for the geometric Satake equivalence
%   \cite{MV}: the fundamental classes of the irreducible components of the intersection
%   $\msf S^{\check\lambda} \cap \overline\Gr_G^{\check\theta}$, the
%   \emph{Mirkovi\'c--Vilonen cycles}, are in bijection with the ``canonical basis'' for
%   the $\check\lambda$-eigenspace of the irreducible $\ch G$-module of lowest weight
%   $\check\theta$.''
%
%   ``The reduced image of the components of $\Gr_B$ in $\Gr_G$ are the semi-infinite orbits
%   $\msf S^{\ch\lambda}(k) = N(F) t^{\ch\lambda} G(\mf o)/G(\mf o)$.''
  \textit{Source: Sakellaridis--Wang (2020), §1.4.}

  For \( \checklambda \in \checkLambda \), the \emph{semi-infinite orbit} is
  \[
    \msf{S}^{\checklambda}(k)
    = N(F)\, t^{\checklambda}\, G(\mfo) / G(\mfo)
    \;\subset\; \Gr_G,
  \]
  where \( \Gr_G = G(F)/G(\mfo) \) is the affine Grassmannian.
  For \( \checktheta \in \checkLambda \) antidominant,
  the \emph{Mirković--Vilonen (MV) cycles} of type \( (\checklambda, \checktheta) \) are the
  irreducible components of
  \( \msf{S}^{\checklambda} \cap \ol{\Gr}_G^{\checktheta} \)
  (the intersection of the semi-infinite orbit with the closure of the
  \( \Lplus G \)-orbit corresponding to \( \checktheta \)).
  By the geometric Satake equivalence, these are in bijection with the canonical basis of
  the \( \checklambda \)-weight space of the irreducible \( \checkG \)-module of lowest
  weight \( \checktheta \).
\end{definition}

\begin{theorem}[IC function formula for affine closure]
  \label{thm:IC-affine-closure}
  \lean{MR4433079IntersectionComplexesAndUnramifiedLFactors.ICFormulaAffineClosure}
  \uses{def:spherical-variety, def:type-T-condition, def:colors-monoid,
        def:IC-function, def:zastava-model, thm:semismall, thm:crystal,
        def:central-fibers}
% SOURCE: [Sakellaridis--Wang 2020], §1.1, Theorem 1.1 (theorem-affine-closure)
%         (read from references/MR4433079-intersection-complexes.tex)
% SOURCE QUOTE:
%   ``\begin{thm}[See \S \ref{sect:finitefields}]
%   \label{theorem-affine-closure}
%   Assume that $X = \overline{H\backslash G}^\aff$ satisfies the conditions above
%   ($T_X=T$, $\check G_X=\check G$ and $\mf c_X \cong \mbb N^r$ is free). Then there is a
%   $(\check T\rtimes\left<\Fr\right>)$-representation $V^+_X$ satisfying:
%   \begin{enumerate}
%   \item the $\check T$-weights of $V^+_X$ belong to $\mathfrak c_X\sm 0$;
%   \item the set of weights of $V^+_X \oplus (V^+_X)^*$ (without multiplicities) equals
%   the set of weights of a $\ch G$-representation $\rho_X$;
%   \item the dimensions of the weight spaces of $V^+_X \oplus (V^+_X)^*$ are invariant
%   under the Weyl group $W$ of $G$;
%   \item the weight spaces in $V^+_X$ of the basis elements $\check\nu_1,\dots,\check\nu_r$
%   of $\mathfrak c_X$ have multiplicity one,
%   \item under the Frobenius action, we have $V^+_X= \bigoplus_{\mB^+} \ol\bbQ_\ell
%   (\smallfrac{1}{2}),$ for some permutation action of $\Fr$ on the multiset $\mB^+$ of
%   weights, compatible with its action on $\ch T$,
%   \end{enumerate}
%   such that the pushforward $\pi_! \Phi_0$ of the IC function satisfies the formula
%   \eqref{equation-pushforward-IC} above.''
  \textit{Source: Sakellaridis--Wang (2020), §1.1, Theorem~1.1.}

  Let \( X = \ol{H \backslash G}^{\mathrm{aff}} \) be the affine closure of an open orbit
  satisfying the type-T condition with \( \mfc_X \cong \mathbb{N}^r \) free, generators
  \( \checknu_1, \ldots, \checknu_r \) (the color valuations).
  Then there exists a \( (\checkT \rtimes \langle \Fr \rangle) \)-representation
  \( V_X^+ \) satisfying:
  \begin{enumerate}
    \item The \( \checkT \)-weights of \( V_X^+ \) lie in \( \mfc_X \setminus \{0\} \).
    \item The set of weights (without multiplicities) of \( V_X^+ \oplus (V_X^+)^* \)
          equals the weight set of a \( \checkG \)-representation \( \rho_X \).
    \item The weight space dimensions of \( V_X^+ \oplus (V_X^+)^* \) are
          \( W \)-invariant.
    \item Each color valuation \( \checknu_i \) appears as a weight of \( V_X^+ \)
          with multiplicity one.
    \item Under the Frobenius action,
          \( V_X^+ \cong \bigoplus_{\mfB^+} \overline{\mathbb{Q}}_\ell(\tfrac{1}{2}) \)
          for a permutation action of \( \Fr \) on the multiset \( \mfB^+ \) of weights
          compatible with its action on \( \checkT \).
  \end{enumerate}
  Moreover, denoting by \( \eta \) the eigencharacter of a \( G \)-eigenmeasure on
  \( X^\bullet(F) \) and by \( \delta = |e^{2\rho_G}| \) the modular character of \( B \),
  the pushforward of the IC function satisfies
  \[
    (\eta\delta)^{1/2} \cdot \pi_! \Phi_0
    = \Bigl(\mathrm{tr}_{\checkT}\!\bigl(\Fr,\, \Sym^\bullet(\check{\mathfrak{n}}(1))\bigr)
      \Bigr)^{-1}
      \cdot \mathrm{tr}_{\checkT}\!\bigl(\Fr,\, \Sym^\bullet(V_X^+)\bigr).
  \]
\end{theorem}

\begin{proof}
  \uses{thm:semismall, thm:crystal, def:compactified-zastava, thm:dimension-estimates}
  The proof occupies §8 of the paper (the ``finite fields'' section).
  The key ingredients are:
  \begin{enumerate}
    \item \emph{Stratified semi-smallness} (\cref{thm:semismall}): the proper map
          \( \bar\pi \colon \ol{\sY}_X \to \sA \) is stratified semi-small, so by the
          decomposition theorem \( \bar\pi_! \IC_{\ol{\sY}} \) is a direct sum of shifted
          IC sheaves.
    \item \emph{Crystal structure and dimension estimates} (\cref{thm:crystal},
          \cref{thm:dimension-estimates}): the irreducible components of
          \( \msf{Y}^{\checklambda} \) of critical dimension biject with the
          \( \checklambda \)-weight space basis of \( V_X^+ \), and their Frobenius-trace
          contributions are \( q^{1/2} \).
    \item \emph{Comparison of Zastava and compactified Zastava} (\cref{def:compactified-zastava}):
          the passage from \( \bar\pi_!\IC_{\ol{\sY}} \) to \( \pi_!\IC_{\sY} \)
          introduces the numerator factor
          \( \prod_{\check\alpha \in \check\Phi^+}(1-q^{-1}e^{\check\alpha}) \).
  \end{enumerate}
  Under the sheaf-function dictionary, tracing Frobenius on stalks gives the stated
  formula for \( \pi_!\Phi_0 \).
\end{proof}

\begin{theorem}[General IC function formula]
  \label{thm:IC-general}
  \lean{MR4433079IntersectionComplexesAndUnramifiedLFactors.ICFormulaGeneral}
  \uses{thm:IC-affine-closure, def:compactified-zastava, def:colors-monoid,
        def:spherical-variety}
% SOURCE: [Sakellaridis--Wang 2020], §1.1, Theorem 1.2 (theorem-general)
%         (read from references/MR4433079-intersection-complexes.tex)
% SOURCE QUOTE:
%   ``\begin{thm}[See \S \ref{sect:finitefields}]
%   \label{theorem-general}
%   In the setting above, if $V_{X^\bullet}^{+}$ denotes the $\check T$-representation for
%   $\overline{X^\bullet}^\aff$ described in Theorem \ref{theorem-affine-closure}, then the
%   pushforward $(\eta\delta)^\frac{1}{2}\cdot \pi_! \Phi_0$ of the IC function for

  \textit{Source: Sakellaridis--Wang (2020), §1.1, Theorem~1.2.}

  In the general setting where \( X^\bullet = H \backslash G \) satisfies the type-T
  condition but \( \mfc_{X^\bullet} \) may not be free, let
  \( \{\checktheta_1, \ldots, \checktheta_d\} = \mathcal{D}^G_{\mathrm{sat}}(X) \)
  be the minimal set of antidominant generators of \( \mfc_X \) not coming from colors.
  For each \( \checktheta_i \), let \( V^{\checktheta_i} \) be the irreducible
  \( \checkG \)-module of lowest weight \( \checktheta_i \).
  Assume the eigencharacter \( \eta \) of the \( G \)-eigenmeasure on \( X(F) \) is of the
  form \( |e^{\mathfrak{h}}| \) for some \( W \)-fixed algebraic character
  \( \mathfrak{h} \in \Lambda^W \).

  Then the IC formula
  \( (\eta\delta)^{1/2} \cdot \pi_!\Phi_0
     = \bigl(\mathrm{tr}_{\checkT}(\Fr, \Sym^\bullet(\check{\mathfrak{n}}(1)))\bigr)^{-1}
       \cdot \mathrm{tr}_{\checkT}(\Fr, \Sym^\bullet(V_X^+)) \)
  holds with
  \[
    V_X^+
    = V_{X^\bullet}^+
      \oplus \bigoplus_{i=1}^d
        V^{\checktheta_i}\!\left(
          \frac{\langle \mathfrak{h} + 2\rho_G,\, \checktheta_i \rangle}{2}
        \right),
  \]
  where \( V_{X^\bullet}^+ \) is the representation from
  \cref{thm:IC-affine-closure} applied to the canonical affine closure
  \( \ol{X^\bullet}^{\mathrm{aff}} \).
\end{theorem}

  \uses{thm:IC-affine-closure, def:compactified-zastava, def:global-model}
  The reduction from Theorem~B to Theorem~A proceeds via the Hecke algebra action on the
  global model \( \sM_X \) (§5 of the paper).
  For each antidominant \( \checktheta_i \in \mfc_X^- \), there is a Hecke correspondence
  \( \sM_{X^{\mathrm{can}}} \times_{\Bun_G} \mathscr{H}^{\checktheta_i}_{G,C} \to \sM_X \)
  that is birational onto its image (Theorem~5.4 of the paper).
  The additional contributions \( V^{\checktheta_i}(\cdots) \) arise from the IC of these
  Hecke strata, which by geometric Satake correspond to the \( \checkG \)-module
  \( V^{\checktheta_i} \).
\end{proof}

\begin{corollary}[Asymptotics of the IC function]
  \label{cor:asymptotics}
  \lean{MR4433079IntersectionComplexesAndUnramifiedLFactors.AsymptoticsICFunction}
  \uses{thm:IC-general, thm:nearby-asymptotics}
% SOURCE: [Sakellaridis--Wang 2020], §1.2, Corollary 1.3 (cor:asymptotics)
%         (read from references/MR4433079-intersection-complexes.tex)
% SOURCE QUOTE:
%   ``\begin{cor}[See \S \ref{sect:asymptotics}] \label{cor:asymptotics}
%   In the setting of Theorem \ref{theorem-general}, we have
%   $(\eta\delta)^\frac{1}{2}(a)  e_\emptyset^* \Phi_0 (N^- a G(\mf o)) =
%   \tr_{\check T}(\Fr, \Sym^\bullet (\check{\mathfrak n}))^{-1} \cdot
%   \tr_{\check T}(\Fr, \Sym^\bullet (V_X^+))$,
%   and $(\eta\delta)^\frac{1}{2}(a)  \Phi_0 (x_0 a G(\mf o)) =

%   \end{cor}''
  \textit{Source: Sakellaridis--Wang (2020), §1.2, Corollary~1.3.}

  In the setting of \cref{thm:IC-general}, let \( e_\emptyset^* \) be the asymptotic map
  \( e_\emptyset^* \colon C^\infty(X^\bullet(F)) \to C^\infty(X_\emptyset^\bullet(F)) \)
  (where \( X_\emptyset \) is the horospherical degeneration of \( X \) with
  \( X_\emptyset^\bullet \cong N^- \backslash G \)).
  For \( a \in T(F) \):
  \[
    (\eta\delta)^{1/2}(a)\; e_\emptyset^* \Phi_0(N^- a G(\mfo))
    =
    \mathrm{tr}_{\checkT}\!\bigl(\Fr,\, \Sym^\bullet(\check{\mathfrak{n}})\bigr)^{-1}
    \cdot \mathrm{tr}_{\checkT}\!\bigl(\Fr,\, \Sym^\bullet(V_X^+)\bigr).
  \]
  Restricting to \( a = t^{\checklambda} \) with
  \( \checklambda \in \checkLambda^{-,\Fr} \) (antidominant, Frobenius-fixed):
  \[
    (\eta\delta)^{1/2}(a)\; \Phi_0(x_0 a G(\mfo))
    =
    \Bigl.\mathrm{tr}_{\checkT}\!\bigl(\Fr,\, \Sym^\bullet(\check{\mathfrak{n}})\bigr)^{-1}
    \cdot \mathrm{tr}_{\checkT}\!\bigl(\Fr,\, \Sym^\bullet(V_X^+)\bigr)
    \Bigr|_{\checkLambda^{-,\Fr}}.
  \]
  The Plancherel decomposition of \( \Phi_0 \) coincides, up to the factor
  \( |W|^{-1} \), with that of \( e_\emptyset^*\Phi_0 \), which by Mellin transform on
  \( X_\emptyset^\bullet \) (with respect to the \( T \)-action) gives the Plancherel
  density
% SOURCE: [Sakellaridis--Wang 2020], §1.2, equation (equation-Plancherel-IC)
%         (read from references/MR4433079-intersection-complexes.tex)
% SOURCE QUOTE:
%   ``\begin{equation}
%    \label{equation-Plancherel-IC}
%   \Vert \Phi_0\Vert^2 = \int_{\hat T/W} \frac{\prod_{\check\alpha\in\check\Phi}
%   (1- e^{\check\alpha}(\chi))}{\prod_{\check\lambda\in S} (1-q^{-\frac{1}{2}}
%   e^{\check\lambda}(\chi))} d\chi = \int_{\hat T/W} L(\chi, V_X, \frac{1}{2})
%   \frac{d\chi}{L(\chi, \check{\mathfrak g}/\check{\mathfrak t},0)}.
%   \end{equation}
%   Here $\mB = \mB^+ \sqcup (-\mB^+)$, and $L(\chi, V_X, 0)$ denotes a local unramified

%   t},0)}$ is the unramified Plancherel measure for $G$.''
  \begin{equation}
    \label{equation-Plancherel-IC}
    \|\Phi_0\|^2
    = \int_{\hat{T}/W}
        \frac{\prod_{\check\alpha \in \check\Phi}(1 - e^{\check\alpha}(\chi))}
             {\prod_{\checklambda \in S}(1 - q^{-1/2} e^{\checklambda}(\chi))}\, d\chi
    = \int_{\hat{T}/W}
        L\!\left(\chi,\, V_X,\, \tfrac{1}{2}\right)
        \frac{d\chi}{L(\chi,\, \checkfg/\check{\mathfrak{t}},\, 0)},
  \end{equation}
  where \( \mfB = \mfB^+ \sqcup (-\mfB^+) \), \( S \) is the corresponding weight multiset,
  \( L(\chi, V_X, \tfrac{1}{2}) \) is a local unramified \( L \)-factor, and the density
  \( \frac{d\chi}{L(\chi, \checkfg/\check{\mathfrak{t}}, 0)} \) is the unramified
  Plancherel measure for \( G \).
\end{corollary}

% SOURCE QUOTE PROOF:
% ``The (function-theoretic) Radon transform $\pi_{\emptyset !}$ on $X_\emptyset^\bullet$ is well-understood;

\begin{proof}
  \uses{thm:IC-general, thm:nearby-asymptotics}
  By \cref{thm:IC-general}, \( (\eta\delta)^{1/2} \cdot \pi_!\Phi_0 \) is expressed as a ratio
  of symmetric algebra traces. The Radon transform \( \pi_{\emptyset!} \) on
  \( X_\emptyset^\bullet \) is well understood: the indicator function
  \( 1_{X_\emptyset^\bullet(\mfo)} \) maps under \( \pi_{\emptyset!} \) to
  \( \delta^{-1/2} \mathrm{tr}_{\checkT}(\Fr, \Sym^\bullet(\check{\mathfrak{n}}))
  \cdot \mathrm{tr}_{\checkT}(\Fr, \Sym^\bullet(\check{\mathfrak{n}}(1)))^{-1} \),

  In the setting of \cref{thm:IC-affine-closure}, when the representation \( \rho_X \)
  predicted by \cref{def:representation-crystal} is \emph{minuscule}
  (i.e.\ all weights of \( \rho_X \) are Weyl-conjugate), Conjecture~1.7 holds:
  \( \mfB_{X^\bullet} \cong B(\rho_X) \), and hence the IC formula of
  \cref{thm:IC-affine-closure} holds with \( \rho_X \) a minuscule symplectic

  In particular, when \( H \) is reductive (so \( H \backslash G \) is affine),
  \( \rho_X \) is always minuscule.
\end{corollary}

  By \cref{thm:crystal}, \( \mfB_{X^\bullet} \) is a semi-normal self-dual crystal with the

\section{Spherical varieties and their arc spaces}

\begin{definition}[Affine spherical \( G \)-variety]
  \label{def:spherical-variety}
  \lean{MR4433079IntersectionComplexesAndUnramifiedLFactors.SphericalVariety}
  \uses{def:reductive-group-data}
% SOURCE: [Sakellaridis--Wang 2020], §2.1 (Spherical varieties) and §1.1
%         (read from references/MR4433079-intersection-complexes.tex)
% SOURCE QUOTE:
%   ``$X$ & an affine $G$-spherical variety over $k$.
%   $X^\circ$ & the open $B$-orbit, for a fixed choice of Borel subgroup $B$.
%   $x_0 \in X^\circ(k)$ & a fixed base point.
%   $H$ & the stabilizer of $x_0$.
%   $X^\bullet$ & the open $G$-orbit $H\bs G$.
%   $X^\can$ & the ``canonical'' affine embedding $\Spec k[X^\bullet]$ of $X^\bullet$.''
  \textit{Source: Sakellaridis--Wang (2020), §2.1 and Notation Table §1.5.}

  An \emph{affine spherical \( G \)-variety} is an affine variety \( X \) over \( k \)
  with a \( G \)-action having an open dense \( B \)-orbit.  We fix:
  \begin{itemize}
    \item An open \( B \)-orbit \( X^\circ \subset X \) and a base point
          \( x_0 \in X^\circ(k) \) with stabilizer \( H \subset G \).
    \item The open \( G \)-orbit \( X^\bullet = H \backslash G \).
    \item The canonical affine embedding \( X^{\mathrm{can}} = \mathrm{Spec}\, k[X^\bullet] \).
    \item The GIT quotient \( X \GIT N = \mathrm{Spec}\, k[X]^N \),
          an affine embedding of a quotient torus \( T_X \) of \( T \).
  \end{itemize}
  Under the type-T condition (\cref{def:type-T-condition}) we will have \( T_X = T \) and
  can identify the open orbit in \( X \GIT N \) with \( T \).
\end{definition}

\begin{definition}[Colors and cocharacter monoid]
  \label{def:colors-monoid}
  \lean{MR4433079IntersectionComplexesAndUnramifiedLFactors.ColorsAndMonoid}
  \uses{def:spherical-variety, def:reductive-group-data}
% SOURCE: [Sakellaridis--Wang 2020], §2 (Spherical varieties) and §1.1
%         (read from references/MR4433079-intersection-complexes.tex)
% SOURCE QUOTE:
%   ``$\mathcal D(X)$ & the set of irreducible $B$-stable divisors in $X$.
%   $\mathcal D$ & the set of colors, i.e., irreducible $B$-stable divisors which are not
%   $G$-stable; equivalently, this can be identified with $\mathcal D(X^\bullet)$.
%   $\varrho_X(D)=\ch\nu_D$ & for $D\in \mathcal D(X)$, the associated $B$-invariant
%   valuation, restricted to the group of nonzero $B$-eigenfunctions:
%   $\check \nu_D: k(X)^{(B)} \to \mathbb Z$, and understood as a functional on the
%   character group $\Lambda_X = k(X)^{(B)}/k^\times$.
%   $\mathfrak c_X^\vee \subset \Lambda_X$ ($\mathfrak c_X\subset \ch\Lambda_X$) &
%   the monoid of weights of $T_X$ on $k[X\sslash N]$ (resp., its dual monoid).''
  \textit{Source: Sakellaridis--Wang (2020), §2 and Notation Table §1.5.}

  The \emph{set of colors} \( \mathcal{D} \) is the set of prime \( B \)-stable divisors of
  the open \( G \)-orbit \( X^\bullet \) (i.e.\ those not \( G \)-stable).
  Each color \( D \in \mathcal{D} \) determines a \( B \)-invariant valuation
  \( \checknu_D \colon k(X)^{(B)} \to \mathbb{Z} \), viewed as an element of
  \( \checkLambda_X \).

  The \emph{cocharacter monoid} \( \mfc_X \subset \checkLambda_X \) is the dual of the
  monoid \( \mfc_X^\vee \) of \( T_X \)-weights on \( k[X \GIT N] \).
  It is a strictly convex submonoid (containing no non-trivial subgroup), and its
  \emph{primitive elements} are \( \Prim(\mfc_X) \).

  We say \( \mfc_X \cong \mathbb{N}^r \) is \emph{free} when the colors
  \( \checknu_1, \ldots, \checknu_r \) (the valuations of the colors) form a
  \( \mathbb{Z} \)-basis for the lattice they generate and freely generate \( \mfc_X \) as a
  monoid.  This is always achievable by passing to a finite abelian cover of \( X^\bullet \)
  and taking its affine closure (§5.2 of the paper).
\end{definition}

  \[

  Under our type-T assumption, \( X^\circ \GIT N \cong T \), so \( \sA \) is the scheme of
  \( \mfc_X \)-valued divisors on \( C \) (see §2 of the paper for the identification of
  \( \sA \) with a partial symmetrization of \( C \) via the primitive generators
  of \( \mfc_X \)).
\end{definition}

\begin{definition}[Compactified Zastava space]
  \label{def:compactified-zastava}
  \lean{MR4433079IntersectionComplexesAndUnramifiedLFactors.CompactifiedZastava}
  \uses{def:zastava-model, def:spherical-variety, def:reductive-group-data}
% SOURCE: [Sakellaridis--Wang 2020], §1.3 and §4

% SOURCE QUOTE:
%   ``one can compactify $\pi$ by considering the \emph{compactified Zastava space}
%   $\barY =   \Maps_\gen(C, (\overline{X/N})/T \supset X^\circ/B),$

  The \emph{compactified Zastava space} is
  \[
    \ol{\sY}_X = \Maps_{\mathrm{gen}}\!\left(C,\;
       \ol{(X/N)}/T \supset X^\circ/B\right),
  \]
  where \( \ol{X/N} \) denotes the stack \( (X \times \ol{N \backslash G})/G \) with
  \( \ol{N \backslash G} = \mathrm{Spec}\, k[N \backslash G] \) the affine closure.

  \[
    \bar\pi \colon \ol{\sY}_X \to \sA
  \]
  extending \( \pi \).
  The difference between \( \bar\pi_! \IC_{\ol{\sY}} \) and \( \pi_! \IC_{\sY} \) accounts
  for the ``numerator'' factor
  \( \prod_{\check\alpha \in \check\Phi^+}(1 - q^{-1} e^{\check\alpha}) \)
  in the IC formula.
  By the decomposition theorem, stratified semi-smallness
  (\cref{thm:semismall}) implies that \( \bar\pi_! \IC_{\ol{\sY}} \) is a direct sum of
  irreducible perverse sheaves on \( \sA \).
\end{definition}


\section{Stratified semi-smallness}

\begin{theorem}[Semi-smallness of compactified Zastava]
  \label{thm:semismall}
  \lean{MR4433079IntersectionComplexesAndUnramifiedLFactors.SemiSmallness}
  \uses{def:compactified-zastava, thm:dimension-estimates,
        def:colors-monoid, def:central-fibers}
% SOURCE: [Sakellaridis--Wang 2020], §1.3, Theorem 1.5 (theorem-semismall)
%         (read from references/MR4433079-intersection-complexes.tex)
% SOURCE QUOTE:
%   ``In Proposition \ref{prop:piproper} and Theorem \ref{thm:semismall} we prove:
%   \begin{thm}
%   \label{theorem-semismall}
%   The map $\bar\pi: \overline{\Scr Y}\to \Scr A$ is proper and stratified semi-small.
%   \end{thm}
%   This is one of the key technical results of this paper, because it allows us to get our
%   hands on the pushforward of the intersection complex, without having a description of
%   the complex itself.
%   The assumption that $\ch G_X=\ch G$ is critical for the theorem: the analogous statement
%   for the usual Finkelberg--Mirkovi\'c Zastava space (\cite{FM, BFGM}) is far from true.''
  \textit{Source: Sakellaridis--Wang (2020), §1.3, Theorem~1.5.}

  The proper map \( \bar\pi \colon \ol{\sY}_X \to \sA \) is \emph{stratified semi-small}
  with respect to the natural stratification of \( \sA \) by strata
  \( \mathring{C}^{\mathfrak{P}} \hookrightarrow \sA \)
  indexed by \( \mfc_X \)-valued divisor types \( \mathfrak{P} \)
  (formal \( \mathbb{N} \)-linear combinations of primitive elements of \( \mfc_X \)).

  That is, for each stratum \( \mathring{C}^{\mathfrak{P}} \) of codimension \( d \) in
  \( \sA \), the fiber of \( \bar\pi \) over a geometric point of
  \( \mathring{C}^{\mathfrak{P}} \) has dimension at most \( d/2 \).
\end{theorem}

% SOURCE QUOTE PROOF:
%   ``Under our assumption that $\mf c_{X^\bullet} = \mbb N^r$, the space $\sY^{\ch\lambda}_{X^\can}$
%   is irreducible. Then the dimension estimate in Theorem~\ref{theorem-slicing}, together
%   with a factorization property of $\sY_X$, implies the ``stratified semi-smallness''
%   condition.''
\begin{proof}
  \uses{thm:dimension-estimates, def:central-fibers, def:compactified-zastava}
  The proof combines two ingredients:
  \begin{enumerate}
    \item \emph{Dimension estimates on central fibers} (\cref{thm:dimension-estimates}):
          for \( X = X^{\mathrm{can}} \), every irreducible component of
          \( \msf{Y}^{\checklambda} \) has dimension at most
          \( \tfrac{1}{2}(\len(\checklambda)-1) \).
    \item \emph{Factorization property of \( \sY_X \)} (§3 and §6 of the paper):
          central fibers over composite strata factor as products of central fibers
          over primitive strata, reducing the semi-smallness check to the single-point
          case \( \mathfrak{P} = [\checklambda] \).
  \end{enumerate}
  Together, these imply that fibers of \( \bar\pi \) over the stratum
  \( \mathring{C}^{[\checklambda]} \) have dimension at most \( \frac{1}{2}(\len(\checklambda)-1) \),
  which equals half the codimension of that stratum, confirming semi-smallness.
\end{proof}

\begin{theorem}[Dimension estimates (slicing theorem)]
  \label{thm:dimension-estimates}
  \lean{MR4433079IntersectionComplexesAndUnramifiedLFactors.DimensionEstimates}
  \uses{def:central-fibers, def:MV-cycles, def:colors-monoid}
% SOURCE: [Sakellaridis--Wang 2020], §1.4, Theorem 1.8 (theorem-slicing)
%         (read from references/MR4433079-intersection-complexes.tex)
% SOURCE QUOTE:
%   ``\begin{thm}
%   \label{theorem-slicing}
%   Let $X = X^\can$ be as in Theorem \ref{theorem-affine-closure}.
%   Let $\msf b$ be an irreducible component of the central fiber $\msf Y^{\check\lambda}$.
%   Then

%   = \check\nu$ for a color $\check\nu$.
%   \end{itemize}
%   \end{thm}''
  \textit{Source: Sakellaridis--Wang (2020), §1.4, Theorem~1.8.}

  Let \( X = X^{\mathrm{can}} \) be as in \cref{thm:IC-affine-closure}.
  Let \( \msf{b} \) be an irreducible component of the central fiber
  \( \msf{Y}^{\checklambda} \).  Then:
  \begin{enumerate}
    \item \( \dim \msf{b} \le \tfrac{1}{2}(\len(\checklambda) - 1) \).
    \item For a color basis element \( \checknu_i \in \mfc_X \),
          \( \msf{Y}^{\checknu_i} = \mathrm{pt} \).
    \item Equality \( \dim \msf{b} = \tfrac{1}{2}(\len(\checklambda)-1) \) holds
          if and only if there exists a sequence of simple roots
          \( \alpha_1, \ldots, \alpha_d \) (with repetitions) such that
          \( \ol{\msf{b}} \cap \msf{S}^{\checklambda - \check{\alpha}_1 - \cdots - \check{\alpha}_j} \)
          has dimension \( \dim \msf{b} - j \) for each \( j \), and
          \( \checklambda - \sum_{i=1}^d \check{\alpha}_i = \checknu \) for some color \( \checknu \).
  \end{enumerate}
\end{theorem}

% SOURCE QUOTE PROOF:
%   ``Our analysis of the central fibers $\msf Y^{\ch\lambda}$ is founded upon the following
%   argument from \cite[\S 3]{MV}. The boundary $\olsf S^{\check\lambda}\sm \msf S^{\check\lambda}
%   = \cup_{\check\nu<\check\lambda} \msf S^{\check\nu}$ is a hyperplane section for some
%   projective embedding of $\Gr_G$. Hence any closed subscheme of $\Gr_G$ which intersects
%   $\olsf S^{\check\lambda}$, also intersects its boundary in codimension one (unless already
%   contained in the boundary). By inductively ``cutting'' by these hyperplanes, we prove:''
\begin{proof}
  \uses{def:MV-cycles, def:central-fibers}
  The proof uses the Mirković--Vilonen boundary hyperplane argument from §3 of [MV]:
  the boundary \( \ol{\msf{S}}^{\checklambda} \setminus \msf{S}^{\checklambda}
  = \bigcup_{\check{\nu} < \checklambda} \msf{S}^{\check{\nu}} \)
  is a hyperplane section of a projective embedding of \( \Gr_G \).
  Hence any closed subscheme meeting \( \ol{\msf{S}}^{\checklambda} \) also meets
  its boundary in codimension one (unless already contained in the boundary).
  Inductively cutting by these boundary hyperplanes reduces \( \checklambda \) to a
  color generator \( \checknu_i \), at each step lowering the dimension bound by one,
  yielding part~(1).
  Part~(2) follows from the identification of \( \msf{Y}^{\checknu_i} \) with a fiber over a
  point on the diagonal stratum \( C \hookrightarrow \sA \) (§5 of the paper).
  Part~(3) characterizes the equality case by tracking the cutting sequence.
\end{proof}


\section{The crystal of a spherical variety}

\begin{theorem}[Crystal structure on \( \mfB_{X^\bullet} \)]
  \label{thm:crystal}
  \lean{MR4433079IntersectionComplexesAndUnramifiedLFactors.CrystalStructure}
  \uses{def:central-fibers, thm:dimension-estimates, def:colors-monoid,
        def:type-T-condition, def:kashiwara-crystal}
% SOURCE: [Sakellaridis--Wang 2020], §1.1 and §7, Theorem 1.6 (theorem-crystal)
%         (read from references/MR4433079-intersection-complexes.tex)
% SOURCE QUOTE:
%   ``\begin{thm}
%   \label{theorem-crystal}
%   Let $X=\overline{X^\bullet}^\aff$ satisfy the assumptions of Theorem
%   \ref{theorem-affine-closure}. The set $\mB_{X^\bullet}$ has the structure of a
%   semi-normal, self-dual crystal over $\ch{\mf g}$ such that the weights have the
%   properties described in Theorem~\ref{theorem-affine-closure}.
%   \end{thm}''
%
%   ``We do not go quite as far in general, but we show that these irreducible components
%   give rise to the aforementioned \emph{crystal}, in the sense of Kashiwara \cite{Kas93},
%   over the Langlands dual Lie algebra $\ch{\mathfrak g}$.''
  \textit{Source: Sakellaridis--Wang (2020), §1.1 and §7, Theorem~1.6.}

  Let \( X = \ol{X^\bullet}^{\mathrm{aff}} \) satisfy the conditions of
  \cref{thm:IC-affine-closure}.
  Let \( \mfB_{X^\bullet} = \mfB_{X^\bullet}^+ \sqcup \mfB_{X^\bullet}^- \), where
  \( \mfB_{X^\bullet}^+ \) is the set of irreducible components of central fibers of
  critical dimension, and \( \mfB_{X^\bullet}^- \) is a formal copy with negated weights.
  Then \( \mfB_{X^\bullet} \) carries the structure of a \emph{semi-normal, self-dual
  Kashiwara crystal} over \( \checkfg \) such that:
  \begin{enumerate}
    \item Weights lie in \( \mfc_X \cup (-\mfc_X) \).
    \item The weight space dimensions are \( W \)-invariant.
    \item The color valuations \( \checknu_i \) appear as weights of \( \mfB_{X^\bullet}^+ \)
          with multiplicity one each.
  \end{enumerate}
\end{theorem}

% SOURCE QUOTE PROOF:
%   ``The construction of the action of the $\SL_2$ corresponding to a simple root $\alpha$
%   of $G$ goes as follows: we factor $X \to X \sslash N$ through
%   $X \to X\sslash N_{P_\alpha} \to X\sslash N$, where $P_\alpha$ is the sub-minimal
%   parabolic corresponding to $\alpha$ and $N_{P_\alpha}$ is its unipotent radical. Then
%   the GIT quotient $X_\alpha:= X\sslash N_{P_\alpha}$ is a spherical variety for the Levi
%   factor $M_\alpha$. But now $X_\alpha$ is (usually) larger than the affine closure of its
%   homogeneous part $X_\alpha^\bullet$. The irreducible components of $\msf Y_X$ of critical
%   dimension (i.e., elements of $\mB_{X^\bullet}^+$) will either go to irreducible
%   components (i) of $\msf Y_{X_\alpha^\bullet}$ of critical dimension or (ii) of
%   $\msf Y_{X_\alpha} \sm \msf Y_{X_\alpha^\bullet}$ (not necessarily of critical
%   dimension).''
\begin{proof}
  \uses{def:central-fibers, thm:dimension-estimates, def:MV-cycles}
  For each simple root \( \alpha \), the crystal operators \( \tilde{e}_\alpha \) and
  \( \tilde{f}_\alpha \) are constructed by factoring
  \( X \to X \GIT N \) through \( X \to X \GIT N_{P_\alpha} \to X \GIT N \)
  (where \( P_\alpha \) is the minimal parabolic for \( \alpha \)),
  and analyzing the map
  \( \mfB_{X^\bullet}^+ \to \) \{irred.\ comps of \( \msf{Y}_{X_\alpha} \) of relevant dim.\}.
  Two cases arise:
  \begin{enumerate}
    \item Components mapping to \( \msf{Y}_{X_\alpha^\bullet} \) of critical dimension
          (the ``induction'' case, reduced to the crystal of a torus torsor over
          \( \mathbb{G}_m \backslash \mathrm{PGL}_2 \)).
    \item Components mapping to the complement, which are shown to be birationally
          equivalent to Mirković--Vilonen cycles
          (\cref{def:MV-cycles}), whose crystal structure is known from [BGcrys].
  \end{enumerate}
  Self-duality is formal (negating weights swaps \( \mfB^+ \) and \( \mfB^- \)).
  Semi-normality (\cref{def:kashiwara-crystal}) follows from the construction of each
  individual \( \mathfrak{sl}_2 \)-component.
  The \( W \)-invariance follows from \cref{thm:dimension-estimates} part~(3).
\end{proof}

\begin{definition}[Kashiwara crystal]
  \label{def:kashiwara-crystal}
  \lean{MR4433079IntersectionComplexesAndUnramifiedLFactors.KashiwaraCrystal}
  \uses{def:reductive-group-data}
% SOURCE: [Sakellaridis--Wang 2020], §7.1 (Definition of crystal),
%         citing Kashiwara 1993 [Kas93]
%         (read from references/MR4433079-intersection-complexes.tex)
% SOURCE QUOTE:
%   ``A crystal $\mB$ over $\ch{\mf g}$ is a set with the following data:
%   $\mathrm{wt} : \mB \to \ch\Lambda_G$
%   $\varepsilon_i, \varphi_i : \mB \to \mbb Z \sqcup \{ -\infty \}$ for $i\in I$,
%   $\tilde e_i, \tilde f_i : \mB \to \mB \sqcup \{0\}$ for $i\in I$,
%   satisfying the following axioms:
%   (1) $\varphi_i(\msf b) = \varepsilon_i(\msf b) + \brac{ \alpha_i, \mathrm{wt}(\msf b)}$
%   for $\msf b \in \mB,\, i\in I$.
%   (2) If $\msf b \in \mB$ and $\tilde e_i \msf b \ne 0$, then
%   $\mathrm{wt}(\tilde e_i \msf b) = \mathrm{wt}(\msf b) + \ch\alpha_i,\;
%   \varepsilon_i( \tilde e_i \msf b) = \varepsilon_i(\msf b) -1,\;
%   \varphi_i( \tilde e_i \msf b) = \varphi_i( \msf b ) + 1$.
%   (3) If $\msf b \in \mB$ and $\tilde f_i \ne 0$, then
%   $\mathrm{wt}(\tilde f_i \msf b) = \mathrm{wt}(\msf b) - \ch\alpha_i,\;
%   \varepsilon(\tilde f_i \msf b) + 1,\;
%   \varphi_i(\tilde f_i \msf b) = \varphi_i(\msf b) -1$.
%   (4) For $\msf b_1, \msf b_2 \in  \mB$, $\msf b_2 = \tilde f_i \msf b_1$ if and only if
%   $\msf b_1 = \tilde e_i \msf b_2$.
%   (5) If $\varphi_i(\msf b)=-\infty$, then $\tilde e_i \msf b = \tilde f_i \msf b = 0$.''
%
%   ``A crystal $\mB$ is called \emph{seminormal}
%   if $\varepsilon_i(\msf b) = \on{max} \{ n \ge 0 \mid \tilde e_i^n \msf b \in \mB \}
%   \in \mathbb N,\quad \varphi_i(\msf b) = \on{max} \{ n \ge 0 \mid \tilde f_i^n \msf b
%   \in \mB \} \in \mathbb N$ for all $\msf b\in \mB, i\in I$.''
  \textit{Source: Sakellaridis--Wang (2020), §7.1, citing Kashiwara (1993).}

  Let \( I \) be the set of vertices of the Dynkin diagram of \( G \), so
  \( \{\alpha_i\}_{i \in I} = \Delta_G \).
  A \emph{Kashiwara crystal} over \( \checkfg \) is a set \( B \) equipped with:
  \begin{itemize}
    \item a weight map \( \mathrm{wt} \colon B \to \checkLambda_G \);
    \item functions
          \( \varepsilon_i, \varphi_i \colon B \to \mathbb{Z} \cup \{-\infty\} \)
          for each \( i \in I \);
    \item raising and lowering operators
          \( \tilde{e}_i, \tilde{f}_i \colon B \to B \cup \{0\} \)
          for each \( i \in I \);
  \end{itemize}
  satisfying the following axioms for all \( \mathsf{b} \in B \) and \( i \in I \):
  \begin{enumerate}
    \item \( \varphi_i(\mathsf{b})
             = \varepsilon_i(\mathsf{b}) + \langle \alpha_i, \mathrm{wt}(\mathsf{b})\rangle \).
    \item If \( \tilde{e}_i \mathsf{b} \ne 0 \):
          \( \mathrm{wt}(\tilde{e}_i \mathsf{b}) = \mathrm{wt}(\mathsf{b}) + \check{\alpha}_i \),
          \( \varepsilon_i(\tilde{e}_i \mathsf{b}) = \varepsilon_i(\mathsf{b}) - 1 \),
          \( \varphi_i(\tilde{e}_i \mathsf{b}) = \varphi_i(\mathsf{b}) + 1 \).
    \item If \( \tilde{f}_i \mathsf{b} \ne 0 \):
          \( \mathrm{wt}(\tilde{f}_i \mathsf{b}) = \mathrm{wt}(\mathsf{b}) - \check{\alpha}_i \),
          \( \varepsilon_i(\tilde{f}_i \mathsf{b}) = \varepsilon_i(\mathsf{b}) + 1 \),
          \( \varphi_i(\tilde{f}_i \mathsf{b}) = \varphi_i(\mathsf{b}) - 1 \).
    \item \( \mathsf{b}_2 = \tilde{f}_i \mathsf{b}_1 \) if and only if
          \( \mathsf{b}_1 = \tilde{e}_i \mathsf{b}_2 \).
    \item If \( \varphi_i(\mathsf{b}) = -\infty \) then
          \( \tilde{e}_i \mathsf{b} = \tilde{f}_i \mathsf{b} = 0 \).
  \end{enumerate}
  The crystal is called \emph{semi-normal} if
  \( \varepsilon_i(\mathsf{b}) = \max\{n \ge 0 \mid \tilde{e}_i^n \mathsf{b} \in B\} \)
  and
  \( \varphi_i(\mathsf{b}) = \max\{n \ge 0 \mid \tilde{f}_i^n \mathsf{b} \in B\} \)
  for all \( \mathsf{b} \in B \) and \( i \in I \).
  It is \emph{self-dual} if it is equipped with an involution
  \( * \colon B \to B \) satisfying
  \( \mathrm{wt}(\mathsf{b}^*) = -\mathrm{wt}(\mathsf{b}) \) and
  swapping the roles of \( \tilde{e}_i \) and \( \tilde{f}_i \).
\end{definition}

  \( \rho_X \), whose irreducible components have highest weights in
  \( \checkLambda^+ \cap W\{\checknu_1, \ldots, \checknu_r\} \).

  (the IC function formula with a symplectic \( \checkG \)-representation \( \rho_X \)).
\end{definition}
