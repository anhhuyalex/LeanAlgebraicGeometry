This chapter states the main formalization targets for the paper \textit{Purity for flat cohomology} by Ce\v{s}navi\v{c}ius and Scholze (MR4681144, arXiv:1912.10932). The goal is to prove that, for a Noetherian local complete intersection ring \((R, \mathfrak{m})\) and a commutative finite flat \(R\)-group scheme \(G\), the flat cohomology groups \(H^i_{\mathfrak{m}}(R, G)\) vanish for \(i < \dim(R)\).

\begin{definition}[Depth with support]

\begin{theorem}[Purity for flat cohomology]
  \label{thm:main}
  \lean{MR4681144PurityForFlatCohomology.Basic.TODO.purityForFlatCohomology}
  \uses{def:ci_local_ring, def:finite_flat_group, def:flat_cohomology_support}
  \textit{Source: Cesnavicius--Scholze, Theorems~1.1 and 1.2.}
  For a Noetherian local ring \((R, \mathfrak{m})\) that is a complete intersection and a commutative finite flat \(R\)-group scheme \(G\),
  \[
    H^i_{\mathfrak{m}}(R, G) = 0 \quad \text{for } \begin{cases} i < \dim(R); \\ i \le \dim(R), \quad \text{if } R \text{ is regular and not a field.} \end{cases}
  \]
\end{theorem}

% \eenum
\begin{proof}
  \uses{thm:abs_coh_pur, thm:perfectoid_purity, thm:excision}
  By \cref{thm:excision}, one reduces to the case where \(R\) is complete. The prime-to-residue-characteristic case (order of \(G\) invertible in \(R/\mathfrak{m}\)) is deduced from the étale purity of \cref{thm:abs_coh_pur} via a Lefschetz hyperplane theorem in local étale cohomology.

  For the \(p\)-primary part with \(p = \operatorname{char}(R/\mathfrak{m}) > 0\): André's lemma produces a flat ind-syntomic perfectoid cover of the completion. The complete intersection hypothesis ensures that the intervening square-zero ideals (arising from killing the regular sequence \(f_1, \ldots, f_c\) that cuts out \(\widehat{R}\) in a regular ring) are free modules and thus of sufficient depth for deformation theory. Excision (\cref{thm:excision}) and the perfectoid purity (\cref{thm:perfectoid_purity}) then give the vanishing.
\end{proof}

\begin{theorem}[Perfectoid purity]
  \label{thm:perfectoid_purity}
  \lean{MR4681144PurityForFlatCohomology.Basic.TODO.perfectoidPurity}
  \uses{def:perfectoid_ring, def:flat_cohomology_support, def:depth_Z}
  \textit{Source: Cesnavicius--Scholze, §1.1.}
  For a perfectoid \(\mathbb{Z}_p\)-algebra \(A\), a commutative finite locally free \(A\)-group scheme \(G\) of \(p\)-power order, and a closed subset \(Z \subset \operatorname{Spec}(A/pA)\) with \(\operatorname{depth}_Z(A) \ge d\),
  \[
    H^i_Z(A, G) = 0 \quad \text{for } i < d.
  \]
\end{theorem}

% SOURCE QUOTE PROOF: For instance, a basic case is when $A$ is a perfect $\bF_p$-algebra. Then, by results of Berthelot \cite{Ber80}, Gabber (unpublished), and Lau \cite{Lau13}, such $A$-groups $G$ are classified by their crystalline Dieudonn\'{e} modules $\bM(G)$, which are $p$-power torsion, finitely presented $W(A)$-modules (that is, $\bA_\Inf(A)$-modules) of projective dimension $\le 1$ equipped with semilinear Frobenius and Verschiebung endomorphisms $F$ and $V$. We use this classification to describe the flat cohomology with coefficients in $G$ in terms of the quasi-coherent cohomology with coefficients in $\bM(G)$: we show in \Cref{KT-input} that
% \be\tag{$1.1.6$} \label{eqn:key-formula}
% R\Gamma_Z(A, G) \cong R\Gamma_Z(\bA_\Inf(A), \bM(G))^{V - 1}.
% \ee
% Since $p$ is a nonzerodivisor in $\bA_\Inf(A)$, the sequence $p, a_1, \dotsc, a_d$ is regular in $\bA_\Inf(A)$ and vanishes on $Z$. By expressing $\bM(G)$ as the cokernel of a map between finite projective $\bA_\Inf(A)$-modules, we may then deduce the vanishing of the right side of \eqref{eqn:key-formula} in the desired range of degrees from the fact that ``enough depth'' implies the vanishing of quasi-coherent cohomology with supports.
\begin{proof}
  \uses{thm:arc_descent}
  The flat cohomology \(R\Gamma_Z(A, G)\) is computed via the key formula
  \[
    R\Gamma_Z(A, G) \cong R\Gamma_Z(\mathbb{A}_{\mathrm{Inf}}(A), \mathbb{M}(G))^{V-1},
  \]
  where \(\mathbb{M}(G)\) is the prismatic Dieudonné module of \(G\). Since \(p\) is a nonzerodivisor in \(\mathbb{A}_{\mathrm{Inf}}(A)\), the sequence \(\xi, a_1, \ldots, a_d\) (where \(\xi\) generates \(\ker(\theta)\)) is regular in \(\mathbb{A}_{\mathrm{Inf}}(A)\) and vanishes on \(Z\). Expressing \(\mathbb{M}(G)\) as the cokernel of a map between finite projective \(\mathbb{A}_{\mathrm{Inf}}(A)\)-modules, the vanishing of quasi-coherent cohomology with supports in the required range gives the result. The key formula is proved using \cref{thm:arc_descent}.
\end{proof}

\begin{corollary}[Gabber's conjecture: Brauer group purity]
  \label{cor:gabber_brauer}
  \lean{MR4681144PurityForFlatCohomology.Basic.TODO.gabberBrauerPurity}
  \uses{thm:main}

  For a Noetherian local ring \((R, \mathfrak{m})\) that is a complete intersection with \(\dim(R) \ge 4\), or with \(R\) regular and \(\dim(R) \ge 2\),
  \[

\end{corollary}

\begin{proof}
  \uses{}
  Follows from \cref{thm:main} applied to group schemes related to \(\mathbb{G}_m\)-torsors and the Kummer sequence.
\end{proof}

\begin{theorem}[Iterated adic continuity]
  \label{thm:adic_continuity}
  \lean{MR4681144PurityForFlatCohomology.Basic.TODO.adicContinuity}
  \uses{def:flat_cohomology_support}
  \textit{Source: Cesnavicius--Scholze, §1.2.}
  For a ring \(R\), an animated \(R\)-algebra \(A\), elements \(a_1, \ldots, a_r \in A\) such that \(A\) agrees with its iterated derived \(a_i\)-adic completion for \(i = 1, \ldots, r\), and a commutative finite locally free \(R\)-group scheme \(G\),
  \[
    R\Gamma(A, G) \xrightarrow{\sim} R\lim_{n > 0} \left( R\Gamma(A/^{\mathbf{L}} (a_1^n, \ldots, a_r^n), G) \right).

\end{proof}

\begin{theorem}[\(p\)-adic continuity]
  \label{thm:p_adic_continuity}
  \lean{MR4681144PurityForFlatCohomology.Basic.TODO.padicContinuity}
  \uses{def:flat_cohomology_support}
  \textit{Source: Cesnavicius--Scholze, §1.2.}
  For a prime \(p\), a ring \(R\), a commutative finite locally free \(R\)-group scheme \(G\) of \(p\)-power order, and an animated \(R\)-algebra \(A\) for which \(\pi_0(A)\) is \(p\)-Henselian,

\subsection{Notation and conventions}

\begin{definition}[Complete intersection local ring]
  \label{def:ci_local_ring}
  \lean{MR4681144PurityForFlatCohomology.Basic.TODO.IsCompleteIntersection}
  \uses{}
% SOURCE: [Cesnavicius--Scholze, arXiv:1912.10932], §1.1 Notation and conventions (read from references/MR4681144-flat-cohomology-source/flat-purity.tex)
% SOURCE QUOTE: All our rings are commutative and unital. We use the definition \cite{EGAIV1}*{chapitre 0, d\'{e}finition~15.1.7, paragraphe 15.2.2} of a regular sequence (so there is no condition on the quotients being nonzero).	 A regular local ring $(R, \fm)$ is \emph{unramified} if it is of mixed characteristic $(0, p)$ and $p \not \in \fm^2$. By the Cohen structure theorem \cite{EGAIV1}*{chapitre 0, th\'{e}or\`{e}me~19.8.8~(i)}, the completion of a Noetherian local ring $(R, \fm)$ is a quotient of a regular ring $\wt{R}$ that may be chosen unramified if  $\Char(R/\fm) = p > 0$. Such an $R$ is a \emph{complete intersection} if the ideal that cuts $\wh{R}$ out in $\wt{R}$ is generated by a regular sequence.
  \textit{Source: Cesnavicius--Scholze, \S1.1 Notation and conventions.}
  A Noetherian local ring \((R, \mathfrak{m})\) is a \emph{complete intersection} if its \(\mathfrak{m}\)-adic completion \(\widehat{R}\) is a quotient of a regular local ring \(\widetilde{R}\) by a regular sequence. Here regular sequences are in the sense of EGA (no condition on the quotients being nonzero). Equivalently, the ideal cutting \(\widehat{R}\) out of \(\widetilde{R}\) is generated by a regular sequence.
\end{definition}

\begin{definition}[Commutative finite flat group scheme]
  \label{def:finite_flat_group}
  \lean{MR4681144PurityForFlatCohomology.Basic.TODO.CommFiniteFlatGroup}
  \uses{}
  A \emph{commutative finite flat \(R\)-group scheme} \(G\) is a finite flat \(R\)-scheme equipped with a commutative group structure over \(\operatorname{Spec}(R)\). Equivalently, \(G\) corresponds to a finite locally free commutative \(R\)-Hopf algebra.
\end{definition}

\begin{definition}[Flat cohomology with support]
  \label{def:flat_cohomology_support}
  \lean{MR4681144PurityForFlatCohomology.Basic.TODO.FlatCohomologyWithSupport}
  \uses{}
  For a ring \(R\), a closed subset \(Z \subset \operatorname{Spec}(R)\), and a commutative finite flat \(R\)-group scheme \(G\), the \emph{flat cohomology with support} \(H^i_Z(R, G)\) is the \(i\)-th cohomology group of the derived functor of sections supported on \(Z\) computed in the fppf topology. When \((R, \mathfrak{m})\) is a local ring and \(Z = \{\mathfrak{m}\}\), we write \(H^i_{\mathfrak{m}}(R, G)\).
\end{definition}

\begin{definition}[Perfectoid ring]
  \label{def:perfectoid_ring}
  \lean{MR4681144PurityForFlatCohomology.Basic.TODO.IsPerfectoid}
  \uses{}
% SOURCE: [Cesnavicius--Scholze, arXiv:1912.10932], §2.1 (read from references/MR4681144-flat-cohomology-source/flat-purity.tex)
% SOURCE QUOTE: \bpp[Perfectoid rings]  \label{perfectoid-def}
% As in \cite{BMS18}*{Definition~3.5}, we say that a ring $A$ is \emph{perfectoid} if
% \benumr
% \item \label{PD-i}
% there is a $\varpi \in A$ with $\varpi^p \mid p$ such that $A$ is $\varpi$-adically (so also $p$-adically) complete, and

% \eenum
  \textit{Source: Cesnavicius--Scholze, §2.1; see also Bhatt--Morrow--Scholze, Definition~3.5.}
  A \(\mathbb{Z}_p\)-algebra \(A\) is \emph{perfectoid} if:
  \begin{enumerate}
    \item there exists \(\varpi \in A\) with \(\varpi^p \mid p\) such that \(A\) is \(\varpi\)-adically (hence also \(p\)-adically) complete; and
    \item the counit map \(\theta \colon \mathbb{A}_{\mathrm{Inf}}(A) \twoheadrightarrow A\) is surjective with principal kernel.
  \end{enumerate}
  Here \(\mathbb{A}_{\mathrm{Inf}}(A) = W(A^{\flat})\) is the Witt vector ring of the tilt \(A^{\flat} = \varprojlim_{a \mapsto a^p} A/pA\).
\end{definition}

\subsection{The \'{e}tale depth is at least the virtual dimension}

\begin{definition}[Virtual dimension]
  \label{def:virtual_dim}
  \lean{MR4681144PurityForFlatCohomology.Basic.TODO.virtualDim}
  \uses{def:ci_local_ring}
% SOURCE: [Cesnavicius--Scholze, arXiv:1912.10932], §3.1 (read from references/MR4681144-flat-cohomology-source/flat-purity.tex)
% SOURCE QUOTE: \bpp[The virtual dimension of a Noetherian local ring] \lab{geom-depth-def}
% For a Noetherian local ring $(R, \fm)$, by Cohen's theorem \cite{EGAIV1}*{chapitre 0, th\'{e}or\`{e}me 19.8.8 (i)}, the $\fm$-adic completion $\wh{R}$ is of the form
% \[%be \lab{Cohen-pres}
% \wh{R} \cong \wt{R}/I \qq \text{for a complete regular local ring} \q \wt{R} \q \text{and an ideal} \q I \subset \wt{R}.
% \]%ee
% The \emph{virtual dimension} of $R$ is %defined by
% \be \label{eqn:vdim-def}
% \vdim(R) \ce \dim(\wt{R}) - (\text{minimal number of generators for $I$})
% \ee
% and, by \cite{SGA2new}*{expos\'{e} XIV, proposition 5.2}, does not depend on the presentation $\wt{R}/I$.
  \textit{Source: Cesnavicius--Scholze, §3.1.}
  Let \((R, \mathfrak{m})\) be a Noetherian local ring. By Cohen's theorem, the \(\mathfrak{m}\)-adic completion \(\widehat{R} \cong \widetilde{R}/I\) for a complete regular local ring \(\widetilde{R}\) and an ideal \(I \subset \widetilde{R}\). The \emph{virtual dimension} of \(R\) is
  \[
    \vdim(R) \coloneqq \dim(\widetilde{R}) - \mu(I),
  \]
  where \(\mu(I)\) is the minimal number of generators of \(I\). This is independent of the Cohen presentation. One has \(\vdim(R) \le \dim(R)\) with equality if and only if \(R\) is a complete intersection.
\end{definition}
