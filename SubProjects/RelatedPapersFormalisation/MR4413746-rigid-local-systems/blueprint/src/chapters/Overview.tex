\chapter{Rigid Local Systems and the Multiplicative Eigenvalue Problem}
\label{chap:overview}

% archon:covers MR4413746RigidLocalSystemsAndTheMultiplicativeEigenvalueProblem.lean MR4413746RigidLocalSystemsAndTheMultiplicativeEigenvalueProblem/Basic.lean

This chapter blueprints the main results of Belkale's paper \emph{Rigid local systems and the multiplicative eigenvalue problem} (MR4413746, arXiv:2005.12457). The central result is a bijection between rigid irreducible unitary local systems on \(\PP^1_\C - S\) with \(n\)-th root of unity local monodromies and the F-vertices of the multiplicative eigenvalue polytope \(P_n(s)\), together with the consequence (Theorem KatzQ) that no such local systems of rank \(>1\) exist when \(n\) is prime.

\section{Rigid local systems}

\begin{definition}[LocalSystem]
  \label{def:local-system}
  \lean{MR4413746RigidLocalSystemsAndTheMultiplicativeEigenvalueProblem.LocalSystem}
  \uses{}
  % SOURCE: Belkale, "Rigid local systems and the multiplicative eigenvalue problem", Section 1.1 (read from references/MR4413746-rigid-local-systems.tex)
  % SOURCE QUOTE: "A rank \ell local system of complex vector spaces on \Bbb{P}^1_{\Bbb{C}}-S where S=\{p_1,\dots,p_s\}, a collection of distinct points, is equivalent to an s-tuple (A_1,\dots,A_s) of matrices A_i\in \op{GL}(\ell,\Bbb{C}) with product A_1A_2\cdots A_s= I_{\ell}, the identity. Such tuples are taken up to conjugacy, i.e., (A_1,\dots,A_s)\sim (CA_1C^{-1}, CA_2C^{-1},\dots, CA_sC^{-1}) where C\in \op{GL}(\ell,\Bbb{C})."
  \textit{Source: Belkale, MR4413746, Section 1.1.}
  A \emph{rank \(\ell\) local system} on \(\PP^1_\C - S\), where \(S = \{p_1, \ldots, p_s\}\) is a finite set of distinct points, is an \(s\)-tuple \((A_1, \ldots, A_s)\) of matrices \(A_i \in \mathrm{GL}(\ell, \C)\) satisfying \(A_1 A_2 \cdots A_s = I_\ell\), taken up to simultaneous conjugacy: \((A_1, \ldots, A_s) \sim (C A_1 C^{-1}, \ldots, C A_s C^{-1})\) for any \(C \in \mathrm{GL}(\ell, \C)\). The conjugacy class of \(A_i\) is called the \emph{local monodromy} at \(p_i\).
\end{definition}

\begin{definition}[IsIrreducible]
  \label{def:irreducible-local-system}
  \lean{MR4413746RigidLocalSystemsAndTheMultiplicativeEigenvalueProblem.IsIrreducible}
  \uses{def:local-system}
  % SOURCE: Belkale, "Rigid local systems and the multiplicative eigenvalue problem", Section 1.1 (read from references/MR4413746-rigid-local-systems.tex)
  % SOURCE QUOTE: "Such a local system is {\em irreducible} if the representation of \pi_1 is irreducible."
  \textit{Source: Belkale, MR4413746, Section 1.1.}
  A local system \((A_1, \ldots, A_s)\) is \emph{irreducible} if the only subspaces of \(\C^\ell\) invariant under all \(A_i\) simultaneously are \(\{0\}\) and \(\C^\ell\).
\end{definition}

\begin{definition}[IsRigid]
  \label{def:rigid-local-system}
  \lean{MR4413746RigidLocalSystemsAndTheMultiplicativeEigenvalueProblem.IsRigid}
  \uses{def:local-system}
  % SOURCE: Belkale, "Rigid local systems and the multiplicative eigenvalue problem", Section 1.1 (read from references/MR4413746-rigid-local-systems.tex)
  % SOURCE QUOTE: "It is called {\em rigid} if any other local system with the same conjugacy classes of local monodromies is isomorphic to it, i.e., if B_1B_2\cdots B_s=I_{\ell} and there exist C_i\in \op{GL}(\ell,\Bbb{C}) such that B_i=C_iA_iC_i^{-1} then there exists a C\in \op{GL}(\ell,\Bbb{C}) such that B_i=CA_iC^{-1} for i=1,\dots,s."
  \textit{Source: Belkale, MR4413746, Section 1.1.}
  A local system \((A_1, \ldots, A_s)\) is \emph{rigid} if any other local system \((B_1, \ldots, B_s)\) with the same conjugacy classes of local monodromies (i.e., there exist \(C_i \in \mathrm{GL}(\ell, \C)\) with \(B_i = C_i A_i C_i^{-1}\)) is globally conjugate to it: there exists \(C \in \mathrm{GL}(\ell, \C)\) such that \(B_i = C A_i C^{-1}\) for all \(i\).
\end{definition}

\begin{definition}[IsUnitary]
  \label{def:unitary-local-system}
  \lean{MR4413746RigidLocalSystemsAndTheMultiplicativeEigenvalueProblem.IsUnitary}
  \uses{def:local-system}
  % SOURCE: Belkale, "Rigid local systems and the multiplicative eigenvalue problem", Section 1.1 (read from references/MR4413746-rigid-local-systems.tex)
  % SOURCE QUOTE: "Any irreducible rigid local system on $\Bbb{P}^1-\{p_1,\dots,p_s\}$ with quasi-unipotent local monodromies (i.e., the $A_i$ have roots of unity as eigenvalues)  is conjugate to the (usual) dual of its complex conjugate, and hence carries a non-degenerate hermitian form \cite[Section 9.5] {Katz}. [...] The case of unitary irreducible rigid local systems, i.e., those for which the form is positive or negative definite, is of special interest because it includes all irreducible rigid local systems with finite global monodromy (with a suitable converse, applied to all Galois conjugates, see e.g., \cite[Theorem 11]{BLocal})."
  \textit{Source: Belkale, MR4413746, Section 1.1.}
  A local system \((A_1, \ldots, A_s)\) is \emph{unitary} if each matrix \(A_i\) is unitary, equivalently if the representation \(\pi_1(\PP^1 - S) \to \mathrm{GL}(\ell, \C)\) factors through \(\mathrm{U}(\ell)\).
\end{definition}

\section{Vertices in the multiplicative eigenvalue problem}

\begin{definition}[EigenvalueSimplex]
  \label{def:delta-n}
  \lean{MR4413746RigidLocalSystemsAndTheMultiplicativeEigenvalueProblem.EigenvalueSimplex}
  \uses{}
  % SOURCE: Belkale, "Rigid local systems and the multiplicative eigenvalue problem", Section 1.2 (read from references/MR4413746-rigid-local-systems.tex)
  % SOURCE QUOTE: "\Delta_n=\Delta(\operatorname{SU}(n))=\{{a}=(a_1,\dots,a_n)\in \Bbb{R}^n\mid a_1\geq a_2\geq\dots \geq a_n\geq a_1-1,\ \sum_{i=1}^n a_i=0\}"
  \textit{Source: Belkale, MR4413746, Section 1.2.}
  The \emph{eigenvalue simplex} for \(\mathrm{SU}(n)\) is the set
  \[
    \Delta_n = \bigl\{ a = (a_1, \ldots, a_n) \in \mathbb{R}^n \mid a_1 \geq a_2 \geq \cdots \geq a_n \geq a_1 - 1,\; \textstyle\sum_{i=1}^n a_i = 0 \bigr\}.
  \]
  Conjugacy classes in \(\mathrm{SU}(n)\) are in bijection with points of \(\Delta_n\): the point \(a \in \Delta_n\) corresponds to the conjugacy class of the diagonal matrix with entries \(\exp(2\pi \sqrt{-1} a_i)\).
\end{definition}

\begin{definition}[MultiplicativeEigenvaluePolytope]
  \label{def:multiplicative-eigenvalue-polytope}
  \lean{MR4413746RigidLocalSystemsAndTheMultiplicativeEigenvalueProblem.MultiplicativeEigenvaluePolytope}
  \uses{def:delta-n}
  % SOURCE: Belkale, "Rigid local systems and the multiplicative eigenvalue problem", Section 1.2 (read from references/MR4413746-rigid-local-systems.tex)
  % SOURCE QUOTE: "Define P_n(s)\subset \Delta_n^s to be the set of tuples \vec{a}=(a^1,a^2,\dots,a^s) such that there exist A_{1}, A_{2}, \dots, A_{s}\in \operatorname{SU}(n) with A_{i} in the conjugacy class corresponding to a^i and A_{1}A_{2}\cdots A_{s}=I_n\in \operatorname{SU}(n)."
  \textit{Source: Belkale, MR4413746, Section 1.2.}
  The \emph{multiplicative eigenvalue polytope} \(P_n(s) \subset \Delta_n^s\) is the set of \(s\)-tuples \(\vec{a} = (a^1, \ldots, a^s) \in \Delta_n^s\) such that there exist matrices \(A_1, \ldots, A_s \in \mathrm{SU}(n)\) with \(A_i\) in the conjugacy class corresponding to \(a^i\) and \(A_1 A_2 \cdots A_s = I_n\). This is a compact convex polytope.
\end{definition}

\begin{lemma}[VerticesOnRegularFaces]
  \label{lem:pinky}
  \lean{MR4413746RigidLocalSystemsAndTheMultiplicativeEigenvalueProblem.vertices_lie_on_regular_faces}
  \uses{def:multiplicative-eigenvalue-polytope, thm:abw}
  % SOURCE: Belkale, "Rigid local systems and the multiplicative eigenvalue problem", Lemma pinky (read from references/MR4413746-rigid-local-systems.tex)
  % SOURCE QUOTE: "Any vertex of $P_{s}(n)$ is on a wall given by equality in one of the inequalities \eqref{wall}, with $\langle\sigma_{I^1}, \sigma_{I^2},\dots,\sigma_{I^s}\rangle_d= 1$."
  \textit{Source: Belkale, MR4413746, Lemma pinky.}
  Every vertex of \(P_n(s)\) lies on a regular facet (i.e., on some wall \(\sum_{j,k \in I^j} a_k^j = d\) with \(\langle \sigma_{I^1}, \ldots, \sigma_{I^s} \rangle_d = 1\)).
\end{lemma}

\begin{remark}[VertexReducible]
  \label{rem:vertex-reducible}
  \uses{def:multiplicative-eigenvalue-polytope, def:unitary-local-system}
  % SOURCE: Belkale, "Rigid local systems and the multiplicative eigenvalue problem", Introduction (read from references/MR4413746-rigid-local-systems.tex)
  % SOURCE QUOTE: "Note that if $\vec{a}\in P_n(s)$ is a vertex and $\rho$ any unitary representation of the fundamental group with local monodromies given by $\vec{a}$, then $\rho$ is reducible (see Lemma \ref{pinky})."
  \textit{Source: Belkale, MR4413746, Introduction (Section 1.2).}
  If \(\vec{a} \in P_n(s)\) is a vertex and \(\rho\) is any unitary representation of the fundamental group \(\pi_1(\PP^1 - S)\) with local monodromies given by \(\vec{a}\), then \(\rho\) is reducible.
\end{remark}

% SOURCE QUOTE PROOF: "If a vertex \vec{a} of P_{s}(n) is not on such a wall, then \vec{a} is a vertex of \Delta_n^s which is in P_{n}(s). Such vertices correspond to scalar matrices \zeta^{m_i}I with \sum m_i a multiple of n, and \zeta a primitive n$th root of unity. Furthermore, any point of \Delta_n^s in a suitable neighbourhood of \vec{a} would again be in P_n(s), but this is a contradiction because the only point in P_{s}(n) with first s-1 coordinates equalling \zeta^{m_i}I, i=1,\dots,s-1 is \vec{a}, since the entries of \vec{a} correspond to central elements of \operatorname{SU}(n)." (read from references/MR4413746-rigid-local-systems.tex, Section 9.4)
\begin{proof}
  \uses{def:multiplicative-eigenvalue-polytope, def:delta-n, thm:abw}
  Suppose \(\vec{a}\) is a vertex of \(P_n(s)\) not lying on any regular facet. Then \(\vec{a}\) must be a vertex of the ambient simplex \(\Delta_n^s\), and since it also lies in \(P_n(s)\), it corresponds to scalar matrices \(\zeta^{m_i} I_n\) with \(\sum m_i \equiv 0 \pmod n\) and \(\zeta = \zeta_n\) a primitive \(n\)-th root of unity. However, any point of \(\Delta_n^s\) in a sufficiently small neighbourhood of \(\vec{a}\) would again lie in \(P_n(s)\) (because the constraints involve only central elements of \(\mathrm{SU}(n)\)), contradicting \(\vec{a}\) being a vertex.
\end{proof}

\begin{definition}[IsFVertex]
  \label{def:f-vertex}
  \lean{MR4413746RigidLocalSystemsAndTheMultiplicativeEigenvalueProblem.IsFVertex}
  \uses{def:multiplicative-eigenvalue-polytope}
  % SOURCE: Belkale, "Rigid local systems and the multiplicative eigenvalue problem", Definition 1.3 (read from references/MR4413746-rigid-local-systems.tex)
  % SOURCE QUOTE: "A vertex \vec{a} of P_n(s) is called a F-vertex if there is a unique unitary local system with local monodromies given by \vec{a}. In other words the corresponding matrix equation A_1A_2\cdots A_s=I_n, A_i\in \op{SU}(n) has a unique (possibly block diagonal) solution up to conjugacy."
  \textit{Source: Belkale, MR4413746, Definition 1.3.}
  A vertex \(\vec{a}\) of \(P_n(s)\) is called an \emph{F-vertex} if there is a unique unitary local system with local monodromies given by \(\vec{a}\), i.e., the matrix equation \(A_1 A_2 \cdots A_s = I_n\) with \(A_i \in \mathrm{SU}(n)\) conjugate to \(a^i\) has a unique solution up to overall conjugacy.
\end{definition}

\section{Parametrization of rigid and irreducible unitary local systems}

\begin{definition}[MonodromyData]
  \label{def:monodromy-data}
  \lean{MR4413746RigidLocalSystemsAndTheMultiplicativeEigenvalueProblem.MonodromyData}
  \uses{}
  % SOURCE: Belkale, "Rigid local systems and the multiplicative eigenvalue problem", Section 1.3 (read from references/MR4413746-rigid-local-systems.tex)
  % SOURCE QUOTE: "Suppose we are given an s-tuple of semisimple conjugacy classes \mathcal{A}=(\bar{A}_1,\dots,\bar{A}_s) in \op{GL}(\ell,\Bbb{C}), which are all n$th roots of the identity (i.e., \bar{A}_i^n=I_{\ell}) such that \prod_{i=1}^s\det \bar{A}_i=1. Assume that the eigenvalues of \bar{A}_i are \exp(2\pi\sqrt{-1}\mu_j^i/n) with integers 0\leq \mu_j^i<n, so that the \mu^i=(\mu^i_1,\dots,\mu^i_{\ell}) are Young diagrams that fit in an \ell\times n box."
  \textit{Source: Belkale, MR4413746, Section 1.3.}
  A \emph{monodromy datum of type \((n, \ell, s)\)} is an \(s\)-tuple \(\mathcal{A} = (\bar{A}_1, \ldots, \bar{A}_s)\) of semisimple conjugacy classes in \(\mathrm{GL}(\ell, \C)\) with \(\bar{A}_i^n = I_\ell\) and \(\prod_{i=1}^s \det \bar{A}_i = 1\). The eigenvalues of \(\bar{A}_i\) are \(\exp(2\pi \sqrt{-1} \mu^i_j / n)\) for integers \(0 \leq \mu^i_j < n\), so that \(\mu^i = (\mu^i_1, \ldots, \mu^i_\ell)\) is a Young diagram fitting in an \(\ell \times n\) box.
\end{definition}

\begin{definition}[StrangeDualityMap]
  \label{def:strange-duality-map}
  \lean{MR4413746RigidLocalSystemsAndTheMultiplicativeEigenvalueProblem.strangeDualityMap}
  \uses{def:monodromy-data, def:delta-n}
  % SOURCE: Belkale, "Rigid local systems and the multiplicative eigenvalue problem", Definition 1.6 (read from references/MR4413746-rigid-local-systems.tex)
  % SOURCE QUOTE: "Consider the vector of transpose Young diagrams \vec{\lambda}=(\lambda^1,\dots,\lambda^s) with \lambda^i=(\mu^i)^T. These are Young diagrams that fit in an n\times\ell box. Define points a^i\in\Delta_{n} as follows: a^i=\frac{1}{\ell}\bigl(\lambda^i -\frac{1}{n}|\lambda^i|(1,1,\dots,1)\bigr)."
  \textit{Source: Belkale, MR4413746, Definition 1.6 and Section 1.3.}
  Given monodromy data \(\mathcal{A}\) as above, form the \emph{transpose Young diagrams} \(\lambda^i = (\mu^i)^T\) (fitting in an \(n \times \ell\) box). Define
  \[
    a^i = \frac{1}{\ell} \Bigl( \lambda^i - \frac{|\lambda^i|}{n}(1, 1, \ldots, 1) \Bigr) \in \Delta_n,
  \]
  where \(|\lambda^i| = \sum_{j=1}^n \lambda^i_j\). The \emph{strange duality map} is \(v(\mathcal{A}) = (a^1, \ldots, a^s) \in \Delta_n^s\).
\end{definition}

\begin{theorem}[Egregium]
  \label{thm:egregium}
  \lean{MR4413746RigidLocalSystemsAndTheMultiplicativeEigenvalueProblem.egregium}
  \uses{def:monodromy-data, def:strange-duality-map, def:f-vertex, def:multiplicative-eigenvalue-polytope, def:rigid-local-system, def:irreducible-local-system, def:unitary-local-system, lem:christmas, lem:fvertex-flinebundle, prop:strange-duality-in-polytope}
  % SOURCE: Belkale, "Rigid local systems and the multiplicative eigenvalue problem", Theorem 1.7 (read from references/MR4413746-rigid-local-systems.tex)
  % SOURCE QUOTE: "Let n be fixed. Then \mathcal{A}\mapsto v(\mathcal{A}) is a bijective correspondence between the following sets: (1) The set of all \ma with varying \ell, but fixed n (i.e., eigenvalues of \bar{A}_i are n$th roots of unity) such that there is a rigid, irreducible unitary local system with local monodromy data \mathcal{A}. (2) The set of F-vertices in P_n(s). Therefore, since there are only finitely many vertices of P_n(s), there are only finitely many unitary irreducible rigid local systems (without fixing the ranks) with eigenvalues of local monodromy n$th roots of unity."
  \textit{Source: Belkale, MR4413746, Theorem 1.7.}
  For fixed \(n\) and \(s = |S|\), the map \(\mathcal{A} \mapsto v(\mathcal{A})\) is a bijective correspondence between:
  \begin{enumerate}
    \item The set of monodromy data \(\mathcal{A}\) (with \(n\) fixed, \(\ell\) varying) for which there exists a rigid irreducible unitary local system on \(\PP^1 - S\) with local monodromy data \(\mathcal{A}\).
    \item The set of F-vertices of \(P_n(s)\).
  \end{enumerate}
  Consequently, since \(P_n(s)\) has only finitely many vertices, there are only finitely many unitary irreducible rigid local systems (across all ranks) with \(n\)-th root of unity local monodromies on \(\PP^1 - S\).
\end{theorem}

\begin{proof}
  \uses{thm:abw, prop:strange-duality-in-polytope, def:f-vertex, lem:pinky, def:f-line-bundle, lem:christmas, lem:fvertex-flinebundle}
  By \cref{lem:christmas}, rigid irreducible unitary local systems on \(\PP^1 - S\) with local monodromy data \(\mathcal{A}\) biject with F-line bundles on \(\mathrm{Par}_{n,\mathcal{O},S}\) of level \(\ell\). By \cref{lem:fvertex-flinebundle}, F-line bundles on \(\mathrm{Par}_{n,\mathcal{O},S}\) biject with F-vertices of \(P_n(s)\). Under these bijections, \(v(\mathcal{A})\) corresponds the F-line bundle to the F-vertex via \cref{prop:strange-duality-in-polytope}. All three steps are sorry-bounded under Route B.
\end{proof}

\begin{corollary}[Existence]
  \label{cor:existence}
  \lean{MR4413746RigidLocalSystemsAndTheMultiplicativeEigenvalueProblem.Existence}
  \uses{def:rigid-local-system, def:irreducible-local-system, def:unitary-local-system, def:gromov-witten-number}
  % SOURCE: Belkale, "Rigid local systems and the multiplicative eigenvalue problem", Corollary existence (read from references/MR4413746-rigid-local-systems.tex)
  % SOURCE QUOTE: "There exists a rigid irreducible unitary local system of rank $\ell$ on $\Bbb{P}^1-S$ whose local monodromy at $p_i$ is conjugate to the diagonal matrix with eigenvalues  $\exp(2\pi\sqrt{-1}\mu_j^i/n)$, $j\in[\ell]$ and $i\in[s]$. Furthermore, the following  equality holds: $$\sum_{i=1}^s \bigl((\ell-\sum_{b=1}^{n-1} c^b_i)^2 \ + \ \sum_{b=1}^{n-1}(c^b_i)^2\bigr)=(s-2)\ell^2+2.$$"
  % SOURCE QUOTE CONTEXT: "Let $\mu^i=(\lambda^i)^T$. These are Young diagrams that fit into an $\ell\times n$ box. [...] $\lambda^i=\sum_{b=1}^{n-1} c_{i}^b\omega_b$, where $\omega_b$ corresponds to $(1,\dots,1,0,\dots,0)$ with $b$ ones and $n-b$ zeros, and is the $b$th fundamental weight."
  \textit{Source: Belkale, MR4413746, Corollary existence.}
  Let \(\mathcal{O}(D(a,j)) = \mathcal{B}(\vec{\lambda}, \ell)\) be the F-line bundle on \(\mathrm{Par}_{n,\mathcal{O},S}\) produced from quantum Schubert calculus, with \(\lambda^i = \sum_{b=1}^{n-1} c^b_i \omega_b\) (where \(\omega_b\) is the \(b\)-th fundamental weight) and \(\mu^i = (\lambda^i)^T\) the transpose Young diagrams fitting in an \(\ell \times n\) box. Then there exists a rigid irreducible unitary local system of rank \(\ell\) on \(\PP^1 - S\) whose local monodromy at \(p_i\) is conjugate to the diagonal matrix with eigenvalues \(\exp(2\pi\sqrt{-1}\,\mu^i_j / n)\) for \(j \in [\ell]\) and \(i \in [s]\). Furthermore, the rigidity identity holds:
  \[
    \sum_{i=1}^s \Bigl( \bigl(\ell - \textstyle\sum_{b=1}^{n-1} c^b_i\bigr)^2 \;+\; \sum_{b=1}^{n-1} (c^b_i)^2 \Bigr) = (s-2)\ell^2 + 2.
  \]
\end{corollary}

\begin{definition}[SymmetryGroupAction]
  \label{def:t-m-map}
  \lean{MR4413746RigidLocalSystemsAndTheMultiplicativeEigenvalueProblem.symmetryGroupAction}
  \uses{def:delta-n}
  % SOURCE: Belkale, "Rigid local systems and the multiplicative eigenvalue problem", Definition 1.10 (read from references/MR4413746-rigid-local-systems.tex)
  % SOURCE QUOTE: "The simplex \Delta_n is identified with \Delta'_n= \{{x}=(x_1,\dots, x_{n-1})\mid x_i\geq 0, \sum x_i=1\}. The correspondence takes (a_1,\dots, a_n) in Definition \ref{defD} to {x} with x_i=a_i-a_{i+1}, i\in[n-1]. If \op{gcd}(m,n)=1, then there is a map T_m:\Delta'_n \to\Delta'_n given by {x}\mapsto {x}' with x'_{mi \pmod{n}}=x_i."
  \textit{Source: Belkale, MR4413746, Definition 1.10.}
  The simplex \(\Delta_n\) is identified with \(\Delta'_n = \{x = (x_1, \ldots, x_{n-1}) \mid x_i \geq 0, \sum x_i = 1\}\) via \(x_i = a_i - a_{i+1}\). For \(\gcd(m, n) = 1\), define \(T_m: \Delta'_n \to \Delta'_n\) by \(x'_{mi \bmod n} = x_i\). This extends to \(T_m: \Delta_n^s \to \Delta_n^s\) coordinate-wise, and permutes vertices of \(P_n(s)\).
\end{definition}

\begin{corollary}[Egregiumprime]
  \label{cor:egregiumprime}
  \lean{MR4413746RigidLocalSystemsAndTheMultiplicativeEigenvalueProblem.egregiumprime}
  \uses{thm:egregium, def:t-m-map, def:f-vertex}
  % SOURCE: Belkale, "Rigid local systems and the multiplicative eigenvalue problem", Corollary 1.11 (read from references/MR4413746-rigid-local-systems.tex)
  % SOURCE QUOTE: "Let n be fixed. Then \mathcal{A}\mapsto v(\mathcal{A}) is a bijective correspondence between the following sets: (1) The set of all \mathcal{A} with varying \ell, but fixed n such that there is a rigid, irreducible local system with local monodromy data \mathcal{A} with finite global monodromy. (2) The set of F-vertices \vec{a} in P_n(s) with the property T_m(\vec{a}) is a F-vertex of P_n(s) for all integers 1\leq m\leq n relatively prime to n."
  \textit{Source: Belkale, MR4413746, Corollary 1.11.}
  The map \(\mathcal{A} \mapsto v(\mathcal{A})\) is a bijection between:
  \begin{enumerate}
    \item Monodromy data \(\mathcal{A}\) (fixed \(n\), varying \(\ell\)) for which there exists a rigid irreducible local system with data \(\mathcal{A}\) and finite global monodromy.
    \item F-vertices \(\vec{a}\) of \(P_n(s)\) such that \(T_m(\vec{a})\) is also an F-vertex for every integer \(m\) with \(1 \leq m \leq n\) and \(\gcd(m, n) = 1\).
  \end{enumerate}
\end{corollary}

\begin{proof}
  \uses{thm:egregium, def:t-m-map}
  The condition on finite global monodromy is equivalent (via Galois theory) to requiring that all Galois conjugates of a given unitary rigid local system are also unitary rigid local systems. Since the Galois group \(\mathrm{Gal}(\mathbb{Q}(\zeta_n)/\mathbb{Q})\) acts via \(\zeta_n \mapsto \zeta_n^m\) (for \(\gcd(m,n)=1\)), and the Galois conjugate corresponds to applying \(T_m\) to the F-vertex, the result follows from Theorem~\cref{thm:egregium}.
\end{proof}

\begin{definition}[TauGroup]
  \label{def:tau-n-s}
  \lean{MR4413746RigidLocalSystemsAndTheMultiplicativeEigenvalueProblem.TauGroup}
  \uses{def:delta-n}
  % SOURCE: Belkale, "Rigid local systems and the multiplicative eigenvalue problem", Definition symmetry (read from references/MR4413746-rigid-local-systems.tex)
  % SOURCE QUOTE: "Use the notation $a\mapsto \alpha\cdot a$ to denote the natural action of $n$th roots of unity $\alpha\in\mu_n=Z(\op{SU}(n))$ on a conjugacy class $a\in \Delta_n$. Define a finite group $\tau_n(s)$ as follows. Here $\zeta=\zeta_n=\exp(\frac{2\pi\sqrt{-1}}{n})\in \Bbb{C}^*$, $$\tau_n(s)=\{(\zeta^{m_1},\zeta^{m_2},\dots,\zeta^{m_s})\mid \sum_{i=1}^s m_i\equiv 0 \pmod{n}\}.$$"
  \textit{Source: Belkale, MR4413746, Definition symmetry.}
  The \emph{symmetry group} \(\tau_n(s)\) is the finite subgroup of \((\mu_n)^s\) defined by
  \[
    \tau_n(s) = \bigl\{ (\zeta^{m_1}, \zeta^{m_2}, \ldots, \zeta^{m_s}) \mid \textstyle\sum_{i=1}^s m_i \equiv 0 \pmod{n} \bigr\},
  \]
  where \(\zeta = \zeta_n = \exp(2\pi\sqrt{-1}/n)\) and \(\mu_n = Z(\mathrm{SU}(n))\) is the center. The group \(\tau_n(s)\) acts coordinate-by-coordinate on \(\Delta_n^s\) and on \(P_n(s)\).
\end{definition}

\begin{lemma}[TauActsFVertex]
  \label{lem:tau-acts-fvertex}
  \lean{MR4413746RigidLocalSystemsAndTheMultiplicativeEigenvalueProblem.tau_acts_fvertex}
  \uses{def:tau-n-s, def:f-vertex, def:multiplicative-eigenvalue-polytope}
  % SOURCE: Belkale, "Rigid local systems and the multiplicative eigenvalue problem", lines following Definition symmetry (read from references/MR4413746-rigid-local-systems.tex)
  % SOURCE QUOTE: "There is a coordinate by coordinate action of $\tau_n(s)$ on $P_n(s)$. The permutation group of $\{1,\dots,s\}$, denoted $S_s$ also acts on $P_n(s)$. The semidirect product $\tau_n(s)\rtimes S_s$ therefore acts on $P_n(s)$ permuting its vertices."
  \textit{Source: Belkale, MR4413746, following Definition symmetry.}
  The group \(\tau_n(s)\) acts coordinate-by-coordinate on \(P_n(s)\), as does the permutation group \(S_s\) of \(\{1, \ldots, s\}\). The semidirect product \(\tau_n(s) \rtimes S_s\) acts on \(P_n(s)\) permuting its vertices, and in particular preserves the set of F-vertices.
\end{lemma}

\begin{proof}
  \uses{def:tau-n-s, def:f-vertex, def:multiplicative-eigenvalue-polytope}
  The coordinate-by-coordinate action of \(\tau_n(s)\) on \(\Delta_n^s\) restricts to an action on \(P_n(s)\) because the conditions defining \(P_n(s)\) are stable under this action. The action permutes vertices and preserves the set of F-vertices. This is sorry-bounded under Route B.
\end{proof}

\begin{theorem}[KatzQ]
  \label{thm:katz-q}
  \lean{MR4413746RigidLocalSystemsAndTheMultiplicativeEigenvalueProblem.katzQ}
  \uses{cor:egregiumprime, prop:property-p, def:rigid-local-system, def:irreducible-local-system, def:tau-n-s, lem:tau-acts-fvertex}
  % SOURCE: Belkale, "Rigid local systems and the multiplicative eigenvalue problem", Theorem 1.8 (read from references/MR4413746-rigid-local-systems.tex)
  % SOURCE QUOTE: "Let n be a prime number. There are no rigid irreducible local systems \mt of rank \ell>1 with finite global monodromy such that the local monodromies of \mt are n$th roots of unity."
  \textit{Source: Belkale, MR4413746, Theorem 1.8.}
  Let \(n\) be a prime number. There are no rigid irreducible local systems of rank \(\ell > 1\) with finite global monodromy such that all local monodromies \(\bar{A}_i\) satisfy \(\bar{A}_i^n = I_\ell\).
\end{theorem}

\begin{proof}
  \uses{prop:property-p, cor:egregiumprime, def:tau-n-s, lem:tau-acts-fvertex}
  By \cref{prop:property-p} part (2), finite global monodromy implies \(\alpha^{-1}\beta\) is not any primitive \(n\)-th root of unity for any two eigenvalues \(\alpha, \beta\) of any \(\bar{A}_i\). When \(n\) is prime, applying the coordinate-wise action of \(\tau_n(s)\) (see \cref{def:tau-n-s}) and permutations via \cref{lem:tau-acts-fvertex}, this forces each local monodromy \(\bar{A}_i\) to be a scalar matrix. A scalar-monodromy local system of rank \(\ell > 1\) is reducible, contradicting irreducibility. Via \cref{cor:egregiumprime}, no F-vertex of \(P_n(s)\) satisfying the finite-monodromy \(T_m\)-stability condition can have rank \(> 1\) when \(n\) is prime.
\end{proof}

\begin{definition}[GromovWittenNumber]
  \label{def:gromov-witten-number}
  \lean{MR4413746RigidLocalSystemsAndTheMultiplicativeEigenvalueProblem.GromovWittenNumber}
  \uses{}
  % SOURCE: Belkale, "Rigid local systems and the multiplicative eigenvalue problem", Definition 1.4 (read from references/MR4413746-rigid-local-systems.tex)
  % SOURCE QUOTE: "The Gromov-Witten number \langle\sigma_{I^1}, \sigma_{I^2},\dots,\sigma_{I^s}\rangle_d is the number of maps (count as zero if infinite) f:\Bbb{P}^1\to \operatorname{Gr}(r,n) of degree d such that for i=1,\dots,s, f(p_i)\in\Omega_{I^j}(F^j_{\bullet})\subseteq \operatorname{Gr}(r,n). Here F^1_{\bullet},\dots,F^s_{\bullet} are complete flags on \Bbb{C}^n in general position."
  \textit{Source: Belkale, MR4413746, Definition 1.4.}
  Let \(I^1, \ldots, I^s\) be subsets of \([n] = \{1, \ldots, n\}\) each of cardinality \(r\), and let \(d \geq 0\) be an integer. Fix distinct points \(p_1, \ldots, p_s \in \PP^1\) and complete flags \(F^1_\bullet, \ldots, F^s_\bullet\) on \(\C^n\) in general position. The \emph{Gromov-Witten number}
  \[
    \langle \sigma_{I^1}, \sigma_{I^2}, \ldots, \sigma_{I^s} \rangle_d
  \]
  is the (finite) count of degree-\(d\) maps \(f: \PP^1 \to \mathrm{Gr}(r, n)\) such that \(f(p_i) \in \Omega_{I^i}(F^i_\bullet)\) for \(i = 1, \ldots, s\), where \(\Omega_{I^i}(F^i_\bullet)\) is the Schubert variety associated to \(I^i\) and \(F^i_\bullet\). This number is independent of the choice of general position flags and of the points \(p_i\).
\end{definition}

\begin{theorem}[ABW]
  \label{thm:abw}
  \lean{MR4413746RigidLocalSystemsAndTheMultiplicativeEigenvalueProblem.abw_inequalities}
  \uses{def:multiplicative-eigenvalue-polytope, def:gromov-witten-number}
  % SOURCE: Belkale, "Rigid local systems and the multiplicative eigenvalue problem", Theorem 1.5 (read from references/MR4413746-rigid-local-systems.tex)
  % SOURCE QUOTE: "Let \vec{a}=(a^1,\dots,a^s)\in \Delta_n^s. Then, \vec{a}\in P_{n}(s) if an only if for any collection of data (d,r,n,I^1,\dots,I^s) as in Definition \ref{GW}, with \langle\sigma_{I^1}, \sigma_{I^2},\dots,\sigma_{I^s}\rangle_d= 1, the inequality \sum_{j=1}^s\sum_{k\in I^j} a_k^{j}\leq d holds."
  \textit{Source: Belkale, MR4413746, Theorem 1.5.}
  A tuple \(\vec{a} = (a^1, \ldots, a^s) \in \Delta_n^s\) belongs to \(P_n(s)\) if and only if for every collection \((d, r, I^1, \ldots, I^s)\) with each \(I^j \subset [n]\) of cardinality \(r\), \(d \geq 0\), and \(\langle \sigma_{I^1}, \ldots, \sigma_{I^s} \rangle_d = 1\), the inequality
  \[
    \sum_{j=1}^s \sum_{k \in I^j} a_k^j \leq d
  \]
  holds. The system of inequalities is irredundant; equality in each defines a \emph{regular facet} of \(P_n(s)\).
\end{theorem}

% SOURCE QUOTE PROOF: (from references/MR4413746-rigid-local-systems.tex, Section 1.4)
% "The system of inequalities in \eqref{wall} is irredundant, see \cite[Remark 8.6]{BKq}, and hence equality in each inequality \eqref{wall} defines a facet of P_n(s). Such facets are called {\em regular}, these are exactly those facets of P_n(s) not contained in one of the alcove walls."
\begin{proof}
  \uses{def:multiplicative-eigenvalue-polytope, def:gromov-witten-number}
  This theorem is due to Agnihotri-Woodward \cite{AW} and Belkale \cite{BLocal}, proved using the Mehta-Seshadri theorem and the theory of parabolic bundles on \(\PP^1\). The key steps are:
  \begin{enumerate}
    \item (Sufficiency) Each inequality \(\sum_{j,k \in I^j} a_k^j \leq d\) with \(\langle \sigma_{I^1}, \ldots, \sigma_{I^s} \rangle_d = 1\) is a necessary condition for the existence of a unitary representation, derived from the Mehta-Seshadri theorem applied to a rank-\(r\) subbundle of degree \(-d\) of the rank-\(n\) bundle associated to the local system.
    \item (Necessity) Conversely, if all the GW inequalities hold, one constructs an \(\mathrm{SU}(n)\)-valued representation directly via the theory of semistable parabolic bundles; the condition \(\langle \sigma \rangle_d = 1\) ensures the requisite subbundles exist.
    \item Irredundancy is proved in \cite[Remark 8.6]{BKq}.
  \end{enumerate}
\end{proof}

  % SOURCE QUOTE: "Let $\mt$ be a rank $\ell$ irreducible rigid local system with semisimple local monodromy data $\mathcal{A}=(\bar{A}_1,\dots,\bar{A}_s)$ in $\op{GL}(\ell,\Bbb{C})$, with $\bar{A}_i^n=1$. Let $i\in[s]$ and let $\alpha$ and $\beta$ be two eigenvalues of $\bar{A}_i$. \begin{enumerate} \item If $\mt$ is unitary, then $\alpha^{-1}\beta\neq \zeta_n$ where $\zeta_n=e^{2\pi\sqrt{-1}/n}$ (i.e., two different eigenvalues of the local monodromy at a point of $S$ cannot be ``as close as possible'' for unitary rigid irreducible local systems). \item If $\mt$ has finite global monodromy, then $\alpha^{-1}\beta$ is not a primitive $n$th root of unity. \end{enumerate}"
  \textit{Source: Belkale, MR4413746, Proposition propP (Section KatzSection).}
  Let \(\mathcal{T}\) be a rank \(\ell\) irreducible rigid local system with \(\bar{A}_i^n = I_\ell\).
  \begin{enumerate}
    \item If \(\mathcal{T}\) is unitary, and \(\alpha, \beta\) are two eigenvalues of some \(\bar{A}_i\), then \(\alpha^{-1}\beta \neq \zeta_n\), where \(\zeta_n = e^{2\pi\sqrt{-1}/n}\). No two eigenvalues of a local monodromy of a unitary rigid irreducible local system can differ by the primitive \(n\)-th root of unity \(\zeta_n\).
    \item If \(\mathcal{T}\) has finite global monodromy, then \(\alpha^{-1}\beta\) is not any primitive \(n\)-th root of unity (i.e., not \(\zeta_n^m\) for any \(m\) with \(\gcd(m,n)=1\)).
  \end{enumerate}
\end{proposition}

\begin{proof}
  \uses{thm:egregium, def:f-vertex, def:tau-n-s}
  Part (1): Via the bijection of \cref{thm:egregium}, the monodromy data \(\mathcal{A}\) corresponds to an F-vertex \(\vec{a} = v(\mathcal{A}) \in P_n(s)\). The F-line bundle \(\mathcal{O}(E)\) on \(\mathrm{Par}_{n,\mathcal{O},S}\) associated to \(\vec{a}\) has the property (from Proposition enumerative of the paper) that if \(\exp(2\pi\sqrt{-1} b/n)\) is an eigenvalue of \(\bar{A}_i\) with multiplicity \(c^b_i > 0\), then \(b \in J^i\) and \(b+1 \notin J^i\) (where \(J^i\) is the set governing the divisor \(E\)). In particular \(1\) and \(\zeta_n\) cannot simultaneously appear as eigenvalues, proving part (1). Part (2): The Galois group \(\mathrm{Gal}(\mathbb{Q}(\zeta_n)/\mathbb{Q})\) acts transitively on the set of primitive \(n\)-th roots of unity, and any Galois conjugate of a unitary rigid irreducible local system with finite global monodromy is again unitary, rigid, and irreducible. Applying part (1) to each Galois conjugate, via the action of \(\tau_n(s)\) from \cref{def:tau-n-s}, yields part (2). This proof is sorry-bounded under Route B.
\end{proof}
