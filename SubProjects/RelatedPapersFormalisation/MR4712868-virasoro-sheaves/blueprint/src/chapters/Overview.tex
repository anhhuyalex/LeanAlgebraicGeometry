\chapter{Overview}
\label{chap:overview}

This chapter gives an overview of the paper ``Virasoro constraints for moduli of sheaves and vertex algebras'' by Bojko--Lim--Moreira (arXiv:2210.05266), which is our formalization target. The paper proves that Virasoro constraints hold for moduli spaces of semistable sheaves on curves and on surfaces with \(h^{1,0} = h^{2,0} = 0\), by expressing these constraints as a primary-state condition inside Joyce's vertex algebra and exploiting wall-crossing compatibility.

The sections below mirror the section (and subsection) structure of the reference paper, and group the formalized items under the paper section from which they originate.

\subsection{Supercommutative algebras}

% SOURCE: [Bojko-Lim-Moreira], Section 2.1 (read from references/MR4712868-virasoro-sheaves.tex)
% SOURCE QUOTE: Before we move onto geometry, we note down some useful observations about freely generated \textit{supercommutative} algebras and derivations on them. Let $D_\bullet$ be a supercommutative $\BZ$-\textit{graded unital algebra} over~$\BC$ with degree $\deg(v) = i$ for any $v\in D_i$. Supercommutativity means that multiplication satisfies
% \[v\cdot w = (-1)^{|v||w|} w\cdot v\,,\]
% where
% $$|v|\in \{0,1\}\,,\quad  \text{such that} \quad |v|\equiv \deg(v)\mod 2\,.$$
% The unit of $D_\bullet$ is always going to be denoted by $1$ and in general, we will omit specifying it in the notation.
% A \textit{superderivation} of degree $r$ on $D_\bullet$ is a $\BZ$-graded linear map
% $$
% R:D_\bullet\longrightarrow D_{\bullet+r}
% $$
% satisfying the \textit{graded Leibnitz rule}
% $$R(v\cdot w) = R(v)\cdot w + (-1)^{r|v|}\,v\cdot R(w)\,.$$

\begin{definition}[SuperCommAlgebra]
  \label{def:supercommutative-algebra}
  \lean{VirasoroSheaves.TODO.SuperCommAlgebra}
  \textit{Source: Bojko--Lim--Moreira, Section 2.1.}
  A \(\mathbb{Z}\)-graded unital \(\mathbb{C}\)-algebra \(D_\bullet\) is \emph{supercommutative}
  if multiplication satisfies
  \[v \cdot w = (-1)^{|v||w|} w \cdot v,\]
  where \(|v| \in \{0,1\}\) is the parity of \(v\), defined by \(|v| \equiv \deg(v) \bmod 2\).

  A \emph{superderivation of degree \(r\)} on \(D_\bullet\) is a graded \(\mathbb{C}\)-linear map
  \(R \colon D_\bullet \to D_{\bullet+r}\) satisfying the graded Leibniz rule:
  \[R(v \cdot w) = R(v) \cdot w + (-1)^{r|v|}\, v \cdot R(w).\]
\end{definition}

% SOURCE: [Bojko-Lim-Moreira], Definition 2.1 and Definition 2.3 (read from references/MR4712868-virasoro-sheaves.tex)
% SOURCE QUOTE: \begin{definition}\label{def: supercommutativefree}
% Let $C_\bullet$ be a $\BZ$-graded  $\BC$-vector space. We denote by $$D_\bullet = \SSym[C_\bullet]$$ the unital supercommutative algebra freely generated by $C_\bullet$. Denote by $C^\bullet$ the graded dual of $C_\bullet$. We define the dual of $D_\bullet$ as a completion of $\SSym[C^\bullet]$ with respect to the degree. More precisely, the dual is
% \[D^\bullet\coloneqq \SSym\llbracket C^\bullet\rrbracket=\prod_{i\geq 0}\SSym[C^\bullet]^i\]
% where $\SSym[C^\bullet]^i$ denotes the degree $i$ part of $\SSym[C^\bullet]$ with the degree induced by the one on $C^\bullet$.
% \end{definition}
% Given a linear map $f\colon C_\bullet\to B_{\bullet+r}$ of degree $r$, there is a unique way to extend $f$ to an algebra homomorphism $\SSym\llbracket f\rrbracket$ and to a derivation $\text{Der}(f)$ of degree $r$.
% [Definition 2.3:] We define a cap product
% $$
% \cap\colon C^\bullet\times D_\bullet\longrightarrow D_\bullet
% $$
% by letting $\nu\cap(-)$ for $\nu\in C^\bullet$ act as a superderivation of degree $-\deg(\nu)$ on $D_\bullet$  restricting to $\langle \nu,-\rangle: C_\bullet\to\BC$. The cap product extends uniquely to
% $$
% \cap\colon D^\bullet\times D_\bullet\longrightarrow D_\bullet\,,
% $$
% by requiring that $(\mu \nu) \cap u = \mu\cap(\nu\cap u)$. Notice that this makes $D_{\bullet}$ into a left $D^\bullet$-module.

\begin{definition}[FreeSuperCommAlgebra]
  \label{def:ssym}
  \lean{VirasoroSheaves.TODO.SSym}
  \uses{def:supercommutative-algebra}
  \textit{Source: Bojko--Lim--Moreira, Definitions 2.1 and 2.3.}
  For a \(\mathbb{Z}\)-graded \(\mathbb{C}\)-vector space \(C_\bullet\), let
  \(\SSym[C_\bullet]\) denote the unital supercommutative algebra freely generated
  by \(C_\bullet\). For \(C^\bullet\) the graded dual of \(C_\bullet\), define the
  dual completion:
  \[D^\bullet \coloneqq \SSym\llbracket C^\bullet \rrbracket
    = \prod_{i \geq 0} \SSym[C^\bullet]^i.\]
  Given any degree-\(r\) linear map \(f \colon C_\bullet \to B_{\bullet+r}\), there
  is a unique algebra homomorphism \(\SSym\llbracket f \rrbracket\) and a unique
  derivation \(\mathrm{Der}(f)\) of degree \(r\) extending \(f\).

  The pairing \(\langle -,- \rangle \colon C^\bullet \times C_\bullet \to \mathbb{C}\)
  extends to a cap product
  \(\cap \colon D^\bullet \times D_\bullet \to D_\bullet\)
  by requiring that \(\nu \cap (-)\) for \(\nu \in C^\bullet\) acts as a
  superderivation of degree \(-\deg(\nu)\) restricting to
  \(\langle \nu, - \rangle\). This makes \(D_\bullet\) a left \(D^\bullet\)-module.
\end{definition}

\begin{definition}[HolDescendentSpace]
  \label{def:ch-space}
  \lean{VirasoroSheaves.TODO.HolDescendentSpace}

\begin{definition}[DescendentAlgebra]
  \label{def:descendent-algebra}
  \lean{VirasoroSheaves.TODO.DescendentAlgebra}

  Write \(\chh_\bullet(\gamma) \coloneqq \sum_{i \geq 0} \chh_i(\gamma) \in \BD^X\).

\begin{definition}[TopDescendentAlgebra]
  \label{def:topological-descendent-algebra}
  \lean{VirasoroSheaves.TODO.TopDescendentAlgebra}
  \uses{def:ssym, def:descendent-algebra}
  \textit{Source: Bojko--Lim--Moreira, Definition 2.7.}
  For a topological type \(\alpha \in K^0_\sst(X)\), let
  \(\widetilde{\mathrm{CH}}^X_\alpha\) be generated by symbols
  \(\mathrm{ch}_i(\gamma)\) for \(i \in \mathbb{Z}_{>0}\), \(\gamma \in H^\bullet(X)\),
  and set \(\BD^X_\alpha \coloneqq \SSym\llbracket \widetilde{\mathrm{CH}}^X_\alpha \rrbracket\).
  There is an algebra homomorphism \(p_\alpha \colon \BD^X \to \BD^X_\alpha\) sending
  \[\chh_i(\gamma) \mapsto \begin{cases}
    \mathrm{ch}_{i - \lfloor (p-q)/2 \rfloor}(\gamma)
      & \text{if } \deg \chh_i(\gamma) > 0, \\
    \int_X \gamma \cdot \mathrm{ch}(\alpha)
      & \text{if } \deg \chh_i(\gamma) = 0, \\
    0 & \text{otherwise.}
  \end{cases}\]
\end{definition}

\subsection{Virasoro operators}

% SOURCE: [Bojko-Lim-Moreira], Section 2.3 (read from references/MR4712868-virasoro-sheaves.tex)
% SOURCE QUOTE: In this section we define the Virasoro operators
% \[\bL_k\colon \BD^X\to \BD^X\quad\textnormal{for}\quad k\geq -1,\]which produce the Virasoro constraints. These operators have two terms, a derivation term $\bR_k$ and a linear term $\bT_k$ which is quadratic in $\chh_i$. The full Virasoro operators are $\bL_k=\bR_k+\bT_k$, where
% \begin{enumerate}
% \item
% $\bR_k \colon \BD^X\to \BD^X$ is an even (of degree $2k$) derivation extended from $\bR_k\colon \tCH^X\to\tCH^X$, where it is defined by
% \[\bR_k \chh_i(\gamma)\coloneqq\left(\prod_{j=0}^k (i+j)\right)\chh_{i+k}(\gamma)\,.\]
% We take the following conventions: the above product is $1$ if $k=-1$ and $\chh_{i+k}(\gamma)=0$ if $i+k<0$.
% \item $\bT_k\colon \BD^X\to \BD^X$ is the operator of multiplication by the element of $\BD^X$ given by
% \[\bT_k\coloneqq\sum_{i+j=k}(-1)^{\dim X-p^L}i!j!\chh_i\chh_j(\td(X)).\]
% \end{enumerate}

\begin{definition}[VirasoroOpR]
  \label{def:virasoro-R}
  \lean{VirasoroSheaves.TODO.VirasoroOpR}
  \uses{def:descendent-algebra}
  \textit{Source: Bojko--Lim--Moreira, Section 2.3, item (1).}
  For \(k \geq -1\), the operator \(\bR_k \colon \BD^X \to \BD^X\) is the even
  superderivation of degree \(2k\) defined on generators by
  \[\bR_k\bigl(\chh_i(\gamma)\bigr) \coloneqq
    \Bigl(\prod_{j=0}^k (i+j)\Bigr)\, \chh_{i+k}(\gamma),\]
  with the convention that the product is \(1\) when \(k = -1\), and
  \(\chh_{i+k}(\gamma) = 0\) if \(i+k < 0\).
\end{definition}

\begin{definition}[VirasoroOpT]
  \label{def:virasoro-T}
  \lean{VirasoroSheaves.TODO.VirasoroOpT}
  \uses{def:descendent-algebra}
  \textit{Source: Bojko--Lim--Moreira, Section 2.3, item (2).}
  For \(k \geq -1\), the operator \(\bT_k \colon \BD^X \to \BD^X\) is
  multiplication by the element
  \[\bT_k \coloneqq \sum_{i+j=k} (-1)^{\dim X - p^L}\, i!\, j!\,
    \chh_i \chh_j(\tdX) \in \BD^X,\]
  where \(\chh_i \chh_j(\tdX) \coloneqq \sum_t (-1)^{\dim X - p_t^L}
  \chh_i(\gamma_t^L)\, \chh_j(\gamma_t^R)\) for the Künneth decomposition
  \(\Delta_* \tdX = \sum_t \gamma_t^L \otimes \gamma_t^R\) with
  \(\gamma_t^L \in H^{p_t^L, q_t^L}(X)\).
\end{definition}

% SOURCE: [Bojko-Lim-Moreira], Section 2.3 (read from references/MR4712868-virasoro-sheaves.tex)
% SOURCE QUOTE: The full Virasoro operators are $\bL_k=\bR_k+\bT_k$, where [items (1) and (2) define $\bR_k$ and $\bT_k$ respectively].
\begin{definition}[VirasoroOp]
  \label{def:virasoro-op}
  \lean{VirasoroSheaves.TODO.VirasoroOp}
  \uses{def:virasoro-R, def:virasoro-T}
  \textit{Source: Bojko--Lim--Moreira, Section 2.3.}
  For \(k \geq -1\), the \emph{Virasoro operator} is
  \[\bL_k \coloneqq \bR_k + \bT_k \colon \BD^X \to \BD^X.\]
\end{definition}

% SOURCE: [Bojko-Lim-Moreira], Remark 2.12 (read from references/MR4712868-virasoro-sheaves.tex)
% SOURCE QUOTE: The operators $\{\bL_k\}_{k\geq -1}$ satisfy the Virasoro bracket relations
% \[[\bL_k, \bL_\ell]=(\ell-k)\bL_{k+\ell}\in \textup{End}(\BD^X)\]
% where the bracket denotes the usual commutator of operators $[X,Y]=X\circ Y-Y\circ X$.
% This was noted in \cite{moop}; for a detailed proof see \cite[Proposition 2.10]{bree}.

\begin{theorem}[VirasoroBracket]
  \label{thm:virasoro-bracket}
  \lean{VirasoroSheaves.TODO.virasoro_bracket}
  \uses{def:virasoro-op}
  \textit{Source: Bojko--Lim--Moreira, Remark 2.12.}
  The operators \(\{\bL_k\}_{k \geq -1}\) satisfy the Virasoro commutator relations
  \[[\bL_k, \bL_\ell] = (\ell - k)\, \bL_{k+\ell} \in \mathrm{End}(\BD^X),\]
  where \([X,Y] = X \circ Y - Y \circ X\) is the commutator.
\end{theorem}

% SOURCE QUOTE PROOF: This was noted in \cite{moop}; for a detailed proof see \cite[Proposition 2.10]{bree}. The unusual constant factor $(\ell-k)$, instead of $(k-\ell)$, suggests that there might be another set of more natural Virasoro operators to which $\{\bL_k\}_{k\geq -1}$ are dual.
\begin{proof}
  \uses{def:virasoro-R, def:virasoro-T}
  Direct computation on generators; see van Bree, Proposition~2.10. The three sub-cases are:
  \begin{enumerate}
  \item[(a)] \([\bR_k, \bR_\ell](\chh_i(\gamma)) = (\ell - k)\bR_{k+\ell}(\chh_i(\gamma))\):
        by substituting the explicit product formula \(\bR_k(\chh_i(\gamma)) = \bigl(\prod_{j=0}^k(i+j)\bigr)\chh_{i+k}(\gamma)\)
        and computing the iterated application \(\bR_k \bR_\ell - \bR_\ell \bR_k\) on generators;
        the result follows from a combinatorial identity on rising factorials.
  \item[(b)] Mixed term \([\bR_k, \bT_\ell]\): since \(\bT_\ell\) is multiplication by
        an element of \(\BD^X\) and \(\bR_k\) is a derivation, the commutator
        \([\bR_k, \bT_\ell](u) = \bR_k(\bT_\ell) \cdot u\) (derivation applied to the
        multiplication element); a direct computation shows \(\bR_k(\bT_\ell) = (\ell - k)\bT_{k+\ell}\)
        up to the correct sign.
  \item[(c)] \([\bT_k, \bT_\ell] = 0\): both \(\bT_k\) and \(\bT_\ell\) are multiplication
        operators on the commutative algebra \(\BD^X\), so their commutator vanishes.
  \end{enumerate}
  Sorry in Lean.
\end{proof}

\begin{conjecture}[VirasoroConjectureNormalized]
  \label{conj:normalizedVirasoro}
  \lean{VirasoroSheaves.TODO.VirasoroConjectureNormalized}
  \uses{def:virasoro-op, def:virasoro-R, def:descendent-algebra}
  \textit{Source: Bojko--Lim--Moreira, Section 2.4 (the \(\delta\)-normalized
  Virasoro conjecture).}
  Let \(M = M_\alpha\) be a moduli space of sheaves of topological type \(\alpha\) and
  let \(\delta \in H^\bullet(X, \mathbb{Z})\) be an algebraic class with
  \(\int_X \delta \cdot \mathrm{ch}(\alpha) \neq 0\). A universal sheaf \(\BG\) is
  \emph{\(\delta\)-normalized} if \(\xi_\BG(\chh_1(\delta)) = 0\). Define the
  \(\delta\)-normalized Virasoro operators (for \(k \geq -1\))
  \[\bL_k^\delta \coloneqq \bR_k + \bT_k + \bS_k^\delta, \qquad
    \bS_k^\delta \coloneqq -\frac{(k+1)!}{\int_X \delta \cdot \mathrm{ch}(\alpha)}\,
    \bR_{-1} \circ \chh_{k+1}(\delta).\]

\begin{definition}[VertexAlgebra]

  \lean{VirasoroSheaves.TODO.VertexAlgebra}
  \textit{Source: Bojko--Lim--Moreira, Section 3.2 (following Kac \cite{Ka98}).}
  A \emph{vertex algebra} is a \(\mathbb{Z}\)-graded \(\mathbb{C}\)-vector space
  \(V_\bullet\) equipped with:
  \begin{enumerate}
  \item a \emph{vacuum vector} \(|0\rangle \in V_0\);
  \item a \emph{translation operator} \(T \colon V_\bullet \to V_{\bullet+2}\);
  \item a \emph{state-field correspondence}

  \begin{enumerate}
  \item \textbf{Vacuum}: \(T|0\rangle = 0\), \(Y(|0\rangle, z) = \mathrm{id}\),

  \item \textbf{Locality}: for all \(a, b \in V_\bullet\), there exists \(N \gg 0\)

  \end{enumerate}
